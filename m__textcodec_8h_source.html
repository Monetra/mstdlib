36 /*! \addtogroup m_textcodec Text Encoding Conversion
37  * \ingroup mstdlib_base
38  *
39  * Text codec converstion. E.g. utf-8 to utf-16.
40  *
41  * utf-8 is used as the base codec. Input for encode should be utf-8 and
42  * output from decode will be utf-8.
43  *
44  * @{
45  */

48 /*! Error handling logic. */
49 typedef enum {
50  M_TEXTCODEC_EHANDLER_FAIL, /*!< Errors should be considered a hard failure. */
51  M_TEXTCODEC_EHANDLER_REPLACE, /*!< Encode replace with ?. Decode replace with U+FFFD. */
52  M_TEXTCODEC_EHANDLER_IGNORE /*!< Ignore data that cannot be encoded or decoded in the codec. */
53 } M_textcodec_ehandler_t;

56 /*! Text codecs that can be used for encoding and decoding. */
57 typedef enum {
58  M_TEXTCODEC_UNKNOWN, /*!< Unknown / invalid codec. */
59  M_TEXTCODEC_UTF8, /*!< Utf-8. */
60  M_TEXTCODEC_ASCII, /*!< Ascii. */
61  M_TEXTCODEC_PERCENT_URL, /*!< Percent with space as %20 for use as a URL rules. Must be utf-8. */
62  M_TEXTCODEC_PERCENT_URLPLUS, /*!< Percent with space as + for use as a URL. Must be utf-8. */
63  M_TEXTCODEC_PERCENT_FORM, /*!< Percent suitable for use as form data. Must be utf-8. */
64  M_TEXTCODEC_CP1252, /*!< Windows code page 1252. */
65  M_TEXTCODEC_ISO88591, /*!< ISO-8859-1. Latin 1. */
66 } M_textcodec_codec_t;

69 /*! Result of a codec conversion. */
70 typedef enum {
71  M_TEXTCODEC_ERROR_SUCCESS, /*!< Successfully converted. */
72  M_TEXTCODEC_ERROR_SUCCESS_EHANDLER, /*!< Succesfully converted based on error handling logic. */
73  M_TEXTCODEC_ERROR_FAIL, /*!< Failure to convert. */
74  M_TEXTCODEC_ERROR_BADINPUT, /*!< Input not in specified encoding. This cannot always be detected and
75  should not be used as a means of determining input encoding. */
76  M_TEXTCODEC_ERROR_INVALID_PARAM, /*!< Invalid parameter. */
77 } M_textcodec_error_t;
