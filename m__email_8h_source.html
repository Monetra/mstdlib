36 /*! \addtogroup m_email Email
37  * \ingroup m_formats
38  *
39  * Email envelope reading and writing.
40  *
41  * This is a flexible implementation and does not auto encode or decode. Also, only
42  * minimal data validation is performed. It is possible generate messages that are
43  * not standard compliant but it should not be possible to generate a message that
44  * with this module that cannot then be parsed by this module.
45  *
46  * Conforms to:
47  * - RFC 5322 Internet Message Format
48  *
49  * Supported:
50  * - RFC 6854 Update to Internet Message Format to Allow Group Syntax in the "From:" and "Sender:" Header Fields
51  *
52  * Not supported:
53  * - RFC 2047 MIME (Multipurpose Internet Mail Extensions) Part Three: Message Header Extensions for Non-ASCII Text
54  * - Splitting multipart within a multipart body part. The sub multipart will be returned as if it is all body data
55  *
56  * There are two types of message parsing supported.
57  * - Stream based callback
58  * - Simple reader (memory buffered)
59  *
60  * Currently supported Read:
61  * - Callback
62  * - Simple
63  *
64  * Currently support Write:
65  * - Simple
66  *
67  * @{
68  */

72 /*! Error codes. */
73 typedef enum {
74  M_EMAIL_ERROR_SUCCESS = 0, /*!< Success. Data fully parsed data is present. More data is possible because email does not have a length indicator. However, a complete message has been seen. */
75  M_EMAIL_ERROR_MOREDATA, /*!< Incomplete message, more data required. Not necessarily an error if parsing as data is streaming. */
76  M_EMAIL_ERROR_STOP, /*!< Stop processing (Used by callback functions to indicate non-error but stop processing). */
77  M_EMAIL_ERROR_INVALIDUSE, /*!< Invalid use. */
78  M_EMAIL_ERROR_HEADER_INVALID, /*!< Header is malformed. */
79  M_EMAIL_ERROR_ADDRESS, /*!< Address is malformed. */
80  M_EMAIL_ERROR_MULTIPART_NOBOUNDARY, /*!< Multipart message missing boundary. */
81  M_EMAIL_ERROR_MULTIPART_HEADER_INVALID, /*!< Multipart message missing boundary. */
82  M_EMAIL_ERROR_MULTIPART_MISSING_DATA, /*!< Multipart data missing. */
83  M_EMAIL_ERROR_MULTIPART_INVALID, /*!< Multipart is invalid. */
84  M_EMAIL_ERROR_NOT_EMAIL, /*!< Not an EMAIL message. */
85  M_EMAIL_ERROR_USER_FAILURE /*!< Generic callback generated failure. */
86 } M_email_error_t;

208 /*! Function definition for reading headers.
209  *
210  * This will provide the full unparsed header.
211  * This is always called for every header.
212  * It may be called multiple times if a header appears multiple times.
213  *
214  * All headers will trigger this function including ones that have
215  * their own dedicated callbacks. If headers are handled in their
216  * resistive dedicated callback, they should be checked for and
217  * ignored when this callback is called.
218  *
219  * A header appearing multiple times here means it was present multiple times.
220  *
221  * \param[in] key Header key.
222  * \param[in] val Header value.
223  * \param[in] thunk Thunk.
224  *
225  * \return Result
226  *
227  * \see M_email_reader_to_func
228  * \see M_email_reader_from_func
229  * \see M_email_reader_cc_func
230  * \see M_email_reader_bcc_func
231  * \see M_email_reader_reply_to_func
232  * \see M_email_reader_subject_func
233  */
234 typedef M_email_error_t (*M_email_reader_header_func)(const char *key, const char *val, void *thunk);

537 /*! Parse email message from given data.
538  *
539  * When a parse returns without error but a full message has not been read, the
540  * parse should be run again starting where the last parse stopped. The reader
541  * can only be used once per complete message.
542  *
543  * Will _not_ return M_EMAIL_ERROR_MOREDATA. It is up to the caller to
