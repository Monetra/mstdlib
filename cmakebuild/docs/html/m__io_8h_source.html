300 /*! Accept an io connection.
301  *
302  * Typically used with network io when a connection is setup as a listening socket.
303  * The io object will remain valid and a new io object for the connection will be created.
304  *
305  * A return of M_IO_ERROR_WOULDBLOCK should not be treated as an error. It means either
306  * there is more data that needs to be received and the event will trigger again. Or
307  * there is no more outstanding connections waiting to be accepted.
308  *
309  * Example:
310  *
311  * void ipserver_listen_callback(M_event_t *el, M_event_type_t type, M_io_t *io, void *thunk)
312  * {
313  * M_io_t *io_out = NULL;
314  * M_io_error_t ioerr;
315  *
316  * (void)thunk;
317  *
318  * if (type != M_EVENT_TYPE_ACCEPT)
319  * return;
320  *
321  * ioerr = M_io_accept(&io_out, io);
322  * if (ioerr != M_IO_ERROR_SUCCESS || io_out == NULL) {
323  * if (ioerr != M_IO_ERROR_WOULDBLOCK) {
324  * // Connection error
325  * }
326  * return;
327  * }
328  *
329  * M_event_add(el, io_out, ipserver_connection_callback, NULL);
330  * }
331  *
332  * \param[out] io_out io object created from the accept.
333  * \param[in] server_io io object which was listening.
334  *
335  * \return Result.
336  */
337 M_API M_io_error_t M_io_accept(M_io_t **io_out, M_io_t *server_io);
