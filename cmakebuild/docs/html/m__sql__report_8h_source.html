46 typedef enum {
47  M_SQL_REPORT_FLAG_NONE = 0, /*!< No special flags */
48  M_SQL_REPORT_FLAG_ALWAYS_ENCAP = 1 << 0, /*!< Always encapsulate the fields, even if there are no conflicting characters.
49  * However, NULL fields will still never be encapsulated so NULL vs empty
50  * strings can be determined by the output. */
51  M_SQL_REPORT_FLAG_OMIT_HEADERS = 1 << 1, /*!< Do not output the first row as headers */
52  M_SQL_REPORT_FLAG_PASSTHRU_UNLISTED = 1 << 2 /*!< By default, all columns in the output report must be specified. This flag
53  will use the column name returned from the SQL server as the header and
54  pass the data thru with no manipulation. Other columns may be overwritten
55  or added, or even suppressed. When set, columns will be output in the order
56  returned from the SQL server, additional added columns will be appended to
57  the end. */
58 } M_sql_report_flags_t;

98 /*! Set desired delimiters, encapsulation, and escaping sequences to be used for the
99  * output data.
100  *
101  * If this function is not called, the defaults are used.
102  *
103  * \param[in] report Initialized report object.
104  * \param[in] field_delim Delimiter to use between fields, default is a comma (,). NULL to not change.
105  * \param[in] field_delim_size Number of characters used in field delimiter. Max size 8.
106  * \param[in] row_delim Delimiter to use between rows, default is a new line (\\r\\n - CRLF). NULL to not change.
107  * \param[in] row_delim_size Number of characters used in row delimiter. Max size 8.
108  * \param[in] field_encaps Encapsulation character to use for field data that may
109  * contain the field_delim or row_delim, default is a double quote ("). NULL to not change.
110  * \param[in] field_encaps_size Number of characters used in field encapsulation. Max size 8.
111  * \param[in] field_escape Escape character to use if the field contains the encapsulation char,
112  * default is the same as the encapsulation, a double quote ("), as this is
113  * what is defined by RFC4180 (CSV)
114  * \param[in] field_escape_size Number of characters used in field escaping. Max size 8.
115  * \return M_TRUE on success, M_FALSE on usage error.
116  */
117 M_API M_bool M_sql_report_set_delims(M_sql_report_t *report, const unsigned char *field_delim, size_t field_delim_size, const unsigned char *row_delim, size_t row_delim_size, const unsigned char *field_encaps, size_t field_encaps_size, const unsigned char *field_escape, size_t field_escape_size);

142 /*! Function callback prototype to use for cell formatting.
143  *
144  * This function signature is used to process every column in a report, the output buffer
145  * is passed in by reference of a certain length. If the buffer is insufficient for the
146  * needs of the formatting function, the caller should allocate a new pointer instead
147  * and set it to the new buffer location. The report generation function will automatically
148  * free the callback's buffer.
149  *
150  * \param[in] stmt Pointer to statement object being processed.
151  * \param[in] arg Custom user-supplied argument for registered callbacks.
152  * \param[in] row Row of result set currently being processed.
153  * \param[in] col Index of result set column being processed, or -1 if no specific column is referenced.
154  * \param[out] buf Pre-allocated #M_buf_t buffer is provided to write column output. The provided buffer is empty,
155  * it is not pre-filled with column data as 'col' may be -1 or otherwise not a 1:1 mapping between
156  * input and output columns. It is up to the person implementing the callback to use the normal
157  * \link m_sql_stmt_result M_sql_stmt_result_*() \endlink functions to get the desired data.
158  * \param[out] is_null Output parameter that if set to M_TRUE, will ignore any contents in buf. It will also prevent
159  * quoting of the output cell so the output differentiates a blank cell (quoted) vs a NULL cell
160  * (unquoted).
161  * \return \return #M_SQL_REPORT_SUCCESS on success or #M_SQL_REPORT_ERROR on failure which will cause report processing to stop
162  * or to just skip the row, return #M_SQL_REPORT_SKIP_ROW.
163  */
164 typedef M_sql_report_cberror_t (*M_sql_report_cell_cb_t)(M_sql_stmt_t *stmt, void *arg, size_t row, ssize_t col, M_buf_t *buf, M_bool *is_null);

225 /*! Process the results from the SQL statement based on the report template configured.
226  *
227  * This function will call the registered report output generation functions to output
228  * each desired column of the report. If row fetching is used due to M_sql_stmt_set_max_fetch_rows(),
229  * this will automatically call M_sql_stmt_fetch() until all rows are returned.
230  *
231  * No state is tracked in the report handle, it may be reused, and used concurrently if
232  * the implementor decides to cache the handle.
233  *
234  * \param[in] report Initialized report object.
235  * \param[in] stmt Executed statement handle.
236  * \param[in] arg Custom user-supplied argument to be passed through to registered callbacks for column formatting.
237  * \param[out] out Formatted report data is returned in this variable and must be free'd by the caller.
238  * \param[out] out_len Optional. Returned length of output data. Should be equivalent to M_str_len(out), but
239  * on huge reports may be more efficient to already know the size.
240  * \param[out] error Buffer to hold error message.
241  * \param[in] error_size Size of error buffer passed in.
242  * \return #M_SQL_ERROR_SUCCESS on success, or one of the #M_sql_error_t error conditions. If an internal error is
243  * generated, the text from the error can be found in the statement handle's error buffer via
244  * M_sql_stmt_get_error_string().
245  */
246 M_API M_sql_error_t M_sql_report_process(const M_sql_report_t *report, M_sql_stmt_t *stmt, void *arg, char **out, size_t *out_len, char *error, size_t error_size);

248 /*! Process a chunk of report data rather than the whole report.
249  *
250  * This function is useful if it is necessary to send a report in pieces either to a file
251  * or via a network connection, especially if the report may become extremely large and
252  * exceed the memory capabilities of the machine.
253  *
254  * This function will be called in a loop until the return value is NOT #M_SQL_ERROR_SUCCESS_ROW,
255  * it will fill in the user-supplied #M_buf_t with the data. It is up to the user to clear
256  * the data from this buffer if the same buffer handle is passed in, or create a new handle,
257  * otherwise data will be appended.
258  *
259  * \warning The caller MUST call this repeatedly until a return value other than #M_SQL_ERROR_SUCCESS_ROW is
260  * returned or otherwise risk memory leaks, or possibly holding a lock on an SQL connection.
261  *
262  * \param[in] report Initialized report object.
263  * \param[in] stmt Executed statement handle.
264  * \param[in] max_rows Maximum number of rows to output per pass. Or 0 to output all. Typically it makes
265  * more sense to just call M_sql_report_process() if you want to use 0 for this value.
266  * \param[in] arg Custom user-supplied argument to be passed through to registered callbacks for column formatting.
267  * \param[in,out] buf User-supplied buffer to append report data to.
268  * \param[in,out] state Pointer to an #M_sql_report_state_t * object, initialized to NULL on first pass. When there
269  * are more rows available, pass the same returned pointer back in. When the report generation
270  * is complete (last pass), this pointer will be automatically cleaned up.
271  * \param[out] error Buffer to hold error message.
272  * \param[in] error_size Size of error buffer passed in.
273  * \return #M_SQL_ERROR_SUCCESS on successful completion of the report, or #M_SQL_ERROR_SUCCESS_ROW if this function
274  * must be called again to get the remaining report data. On failure, one of the #M_sql_error_t errors may
275  * be returned
276  */
277 M_API M_sql_error_t M_sql_report_process_partial(const M_sql_report_t *report, M_sql_stmt_t *stmt, size_t max_rows, void *arg, M_buf_t *buf, M_sql_report_state_t **state, char *error, size_t error_size);
