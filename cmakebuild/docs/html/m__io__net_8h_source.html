35 /*! \addtogroup m_io_net Network I/O
36  * \ingroup m_eventio_base
37  *
38  * Network I/O functions
39  *
40  * Capable of functioning as a network server and client.
41  *
42  * Examples
43  * ========
44  *
45  * Client
46  * ------
47  *
48  * Example network client which downloads google.com's home page. This uses
49  * a network client connection wrapped in TLS. A trace layer is provided and
50  * is commented out.
51  *
52  * \code{.c}
53  * #include <mstdlib/mstdlib.h>
54  * #include <mstdlib/mstdlib_io.h>
55  * #include <mstdlib/mstdlib_tls.h>
56  *
57  * static void do_trace(void *cb_arg, M_io_trace_type_t type, M_event_type_t event_type, const unsigned char *data, size_t data_len)
58  * {
59  * char *out;
60  *
61  * switch (type) {
62  * case M_IO_TRACE_TYPE_READ:
63  * M_printf("READ:\n");
64  * break;
65  * case M_IO_TRACE_TYPE_WRITE:
66  * M_printf("WRITE:\n");
67  * break;
68  * default:
69  * return;
70  * }
71  * out = M_str_hexdump(M_STR_HEXDUMP_HEADER, 0, "\t", data, data_len);
72  * M_printf("%s\n", out);
73  * M_free(out);
74  * }
75  *
76  * static void run_cb(M_event_t *el, M_event_type_t etype, M_io_t *io, void *thunk)
77  * {
78  * M_buf_t *connected_buf = thunk;
79  * char buf[128] = { 0 };
80  * size_t len_written = 0;
81  *
82  * switch (etype) {
83  * case M_EVENT_TYPE_CONNECTED:
84  * M_printf("CONNECTED: %s%s%s:%d (%s: %s - session%sreused)\n",
85  * M_io_net_get_type(io)==M_IO_NET_IPV6?"[":"",
86  * M_io_net_get_ipaddr(io),
87  * M_io_net_get_type(io)==M_IO_NET_IPV6?"]":"",
88  * M_io_net_get_port(io),
89  * M_tls_protocols_to_str(M_tls_get_protocol(io, M_IO_LAYER_FIND_FIRST_ID)),
90  * M_tls_get_cipher(io, M_IO_LAYER_FIND_FIRST_ID),
91  * M_tls_get_sessionreused(io, M_IO_LAYER_FIND_FIRST_ID)?"":" not ");
92  *
93  * M_io_write_from_buf(io, connected_buf);
94  * break;
95  * case M_EVENT_TYPE_READ:
96  * M_io_read(io, (unsigned char *)buf, sizeof(buf), &len_written);
97  * if (len_written > 0) {
98  * M_printf("%.*s", (int)len_written, buf);
99  * }
100  * break;
101  * case M_EVENT_TYPE_WRITE:
102  * M_io_write_from_buf(io, connected_buf);
103  * break;
104  * case M_EVENT_TYPE_DISCONNECTED:
105  * case M_EVENT_TYPE_ERROR:
106  * M_event_done_with_disconnect(el, 1000);
107  * break;
108  * case M_EVENT_TYPE_ACCEPT:
109  * case M_EVENT_TYPE_OTHER:
110  * M_event_done(el);
111  * break;
112  * }
113  * }
114  *
115  * int main(int argc, char *argv)
116  * {
117  * M_event_t *el;
118  * M_dns_t *dns;
119  * M_buf_t *buf;
120  * M_io_t *io;
121  * M_tls_clientctx_t *ctx;
122  * size_t layer_id;
123  *
124  *
125  * dns = M_dns_create();
126  * el = M_event_create(M_EVENT_FLAG_NONE);
127  * buf = M_buf_create();
128  * M_buf_add_str(buf, "GET / HTTP/1.1\r\n");
129  * M_buf_add_str(buf, "Host: www.google.com\r\n");
130  * M_buf_add_str(buf, "Connection: close\r\n");
131  * M_buf_add_str(buf, "\r\n");
132  *
133  * M_io_net_client_create(&io, dns, "google.com", 443, M_IO_NET_ANY);
134  * ctx = M_tls_clientctx_create();
135  * M_tls_clientctx_set_default_trust(ctx);
136  * M_io_tls_client_add(io, ctx, NULL, &layer_id);
137  * M_tls_clientctx_destroy(ctx);
138  * //M_io_add_trace(io, &layer_id, do_trace, NULL, NULL, NULL);
139  *
140  * M_event_add(el, io, run_cb, buf);
141  * M_event_loop(el, M_TIMEOUT_INF);
142  *
143  * M_io_destroy(io);
144  * M_buf_cancel(buf);
145  * M_event_destroy(el);
146  * M_dns_destroy(dns);
147  * return 0;
148  * }
149  * \endcode
150  *
151  * Server
152  * ------
153  *
154  * Example network sever. This is an echo server which uses a state machine
155  * to determine what operation it should perform.
156  *
157  * \code{.c}
158  * #include <mstdlib/mstdlib.h>
159  * #include <mstdlib/mstdlib_io.h>
160  *
161  * typedef enum {
162  * STATE_CHECK = 1,
163  * STATE_ECHO,
164  * STATE_EXIT
165  * } states_t;
166  *
167  * typedef struct {
168  * M_buf_t *write_buf;
169  * M_parser_t *read_parser;
170  * M_io_t *io;
171  * M_event_t *el;
172  * M_state_machine_t *sm;
173  * } ldata_t;
174  *
175  * static M_state_machine_status_t state_check(void *data, M_uint64 *next)
176  * {
177  * ldata_t *ldata = data;
178  *
179  * (void)next;
180  *
181  * M_parser_mark(ldata->read_parser);
182  * if (M_parser_len(ldata->read_parser) == 0 || M_parser_consume_until(ldata->read_parser, (const unsigned char *)"\n", 1, M_TRUE) == 0) {
183  * M_parser_mark_rewind(ldata->read_parser);
184  * return M_STATE_MACHINE_STATUS_WAIT;
185  * }
186  *
187  * return M_STATE_MACHINE_STATUS_NEXT;
188  * }
189  *
190  * static M_state_machine_status_t state_echo(void *data, M_uint64 *next)
191  * {
192  * ldata_t *ldata = data;
193  * char *out;
194  *
195  * out = M_parser_read_strdup_mark(ldata->read_parser);
196  * M_buf_add_str(ldata->write_buf, out);
197  * M_io_write_from_buf(ldata->io, ldata->write_buf);
198  *
199  * if (!M_str_eq(out, "EXIT\r\n") && !M_str_eq(out, "EXIT\n"))
200  * *next = STATE_CHECK;
201  *
202  * M_free(out);
203  * return M_STATE_MACHINE_STATUS_NEXT;
204  * }
205  *
206  * static M_state_machine_status_t state_exit(void *data, M_uint64 *next)
207  * {
208  * ldata_t *ldata = data;
209  * (void)next;
210  * M_event_done_with_disconnect(ldata->el, 1000);
211  * }
212  *
213  * static void connection_cb(M_event_t *el, M_event_type_t etype, M_io_t *io, void *thunk)
214  * {
215  * ldata_t *ldata = thunk;
216  * char error[256] = { 0 };
217  * M_bool clean = M_FALSE;
218  * M_state_machine_status_t status;
219  *
220  * switch (etype) {
221  * case M_EVENT_TYPE_CONNECTED:
222  * M_printf("CLIENT CONNECTED: %s%s%s:%d\n",
223  * M_io_net_get_type(io)==M_IO_NET_IPV6?"[":"",
224  * M_io_net_get_ipaddr(io),
225  * M_io_net_get_type(io)==M_IO_NET_IPV6?"]":"",
226  * M_io_net_get_port(io));
227  * break;
228  * case M_EVENT_TYPE_READ:
229  * if (M_io_read_into_parser(io, ldata->read_parser) == M_IO_ERROR_SUCCESS) {
230  * status = M_state_machine_run(ldata->sm, ldata);
231  * if (status != M_STATE_MACHINE_STATUS_WAIT) {
232  * M_io_disconnect(io);
233  * }
234  * }
235  * break;
236  * case M_EVENT_TYPE_WRITE:
237  * if (M_buf_len(ldata->write_buf) > 0) {
238  * M_io_write_from_buf(io, ldata->write_buf);
239  * }
240  * break;
241  * case M_EVENT_TYPE_DISCONNECTED:
242  * clean = M_TRUE;
243  * case M_EVENT_TYPE_ACCEPT:
244  * case M_EVENT_TYPE_ERROR:
245  * case M_EVENT_TYPE_OTHER:
246  * if (!clean) {
247  * M_io_get_error_string(io, error, sizeof(error));
248  * }
249  * M_printf("CLIENT DISCONNECTED: %s%s%s:%d (%s%s%s)\n",
250  * M_io_net_get_type(io)==M_IO_NET_IPV6?"[":"",
251  * M_io_net_get_ipaddr(io),
252  * M_io_net_get_type(io)==M_IO_NET_IPV6?"]":"",
253  * M_io_net_get_port(io),
254  * clean?"clean":"unclean", clean?"":" - ", clean?"":error);
255  *
256  * M_io_destroy(io);
257  * M_state_machine_destroy(ldata->sm);
258  * M_buf_cancel(ldata->write_buf);
259  * M_parser_destroy(ldata->read_parser);
260  * M_free(ldata);
261  * break;
262  * }
263  * }
264  *
265  * static void listen_cb(M_event_t *el, M_event_type_t etype, M_io_t *io, void *thunk)
266  * {
267  * M_io_t *io_out = NULL;
268  * ldata_t *ldata;
269  * M_io_error_t ioerr;
270  * char error[256] = { 0 };
271  *
272  * (void)thunk;
273  *
274  * switch (etype) {
275  * case M_EVENT_TYPE_ACCEPT:
276  * ioerr = M_io_accept(&io_out, io);
277  * if (ioerr == M_IO_ERROR_WOULDBLOCK) {
278  * return;
279  * } else if (ioerr != M_IO_ERROR_SUCCESS || io_out == NULL) {
280  * M_io_get_error_string(io, error, sizeof(error));
281  * M_printf("ACCEPT FAILURE: %s\n", error);
282  * M_io_destroy(io_out);
283  * }
284  *
285  * ldata = M_malloc_zero(sizeof(*ldata));
286  * ldata->el = el;
287  * ldata->write_buf = M_buf_create();
288  * ldata->read_parser = M_parser_create(M_PARSER_FLAG_NONE);
289  * ldata->io = io_out;
290  * ldata->sm = M_state_machine_create(0, NULL, M_STATE_MACHINE_NONE);
291  * M_state_machine_insert_state(ldata->sm, STATE_CHECK, 0, NULL, state_check, NULL, NULL);
292  * M_state_machine_insert_state(ldata->sm, STATE_ECHO, 0, NULL, state_echo, NULL, NULL);
293  * M_state_machine_insert_state(ldata->sm, STATE_EXIT, 0, NULL, state_exit, NULL, NULL);
294  *
295  * M_event_add(el, io_out, connection_cb, ldata);
296  * break;
297  * case M_EVENT_TYPE_CONNECTED:
298  * case M_EVENT_TYPE_READ:
299  * case M_EVENT_TYPE_WRITE:
300  * break;
301  * case M_EVENT_TYPE_DISCONNECTED:
302  * case M_EVENT_TYPE_ERROR:
303  * case M_EVENT_TYPE_OTHER:
304  * M_io_destroy(io);
305  * break;
306  * }
307  * }
308  *
309  * int main(int argc, char *argv)
310  * {
311  * M_event_t *el;
312  * M_io_t *io = NULL;
313  * M_io_error_t ioerr;
314  *
315  *
316  * ioerr = M_io_net_server_create(&io, 8999, NULL, M_IO_NET_IPV4);
317  * if (ioerr != M_IO_ERROR_SUCCESS) {
318  * M_printf("Could not start server: %s\n", M_io_error_string(ioerr));
319  * return 0;
320  * }
321  *
322  * el = M_event_create(M_EVENT_FLAG_NONE);
323  *
324  * M_event_add(el, io, listen_cb, NULL);
325  * M_event_loop(el, M_TIMEOUT_INF);
326  *
327  * M_event_destroy(el);
328  * return 0;
329  * }
330  * \endcode
331  *
332  * @{
333  */
