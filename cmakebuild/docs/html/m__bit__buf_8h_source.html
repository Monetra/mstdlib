37 /*! \addtogroup m_bit_buf Bitwise Buffered Data Builder
38  * \ingroup mstdlib_base
39  *
40  * Allows for buffered writing of data that's organized per-bit, instead of per-byte.
41  *
42  * When done adding data the contents of the buffer can be output as a continuous byte-array, either
43  * raw as an (M_uint8 *), or inside a regular per-byte M_buf_t.
44  *
45  * Example (creating a buffer, adding data, finishing the buffer):
46  *
47  * \code{.c}
48  * M_bit_buf_t *bbuf;
49  * M_uint8 *out;
50  * size_t out_nbytes;
51  *
52  * bbuf = M_bit_buf_create();
53  * M_bbuf_add_bit(bbuf, 1);
54  * M_bbuf_add(bbuf, 0xA2C4, 14); // adds least-significant 14 bits of 0xA2C4
55  * M_bbuf_add_str(bbuf, "100010000"); // adds 9 bits from binary-ASCII
56  *
57  * out = M_bit_buf_finish(bbuf, &out_nbytes);
58  *
59  * // out now points to a byte buffer containing 3 bytes. Now can output to disk, do further processing, etc.
60  *
61  * M_free(out);
62  * \endcode
63  *
64  * @{
65  */

70 /*! Enumeration for byte-alignment padding mode for M_bit_buf_add().*/
71 typedef enum {
72  M_BIT_BUF_PAD_NONE = 0, /*!< Don't add any padding. */

74  M_BIT_BUF_PAD_AFTER /*!< Pad with zero bits after new value, so that bit stream after add is byte-aligned */
75 } M_bit_buf_pad_t;

142 /*! Return pointer to internal buffer data.
143  *
144  * The internal data is stored as an array of bytes. The first bit in the buffer is always
145  * guaranteed to be the highest bit in the first byte, second bit in the buffer is the next
146  * highest bit, and so on. The implementation guarantees this will always be the case,
147  * regardless of what operations you may have done on the bit buffer.
148  *
149  * \warning
150  * The returned pointer may be invalidated when you add data to the buffer. For safety,
151  * it should be used immediately after you call M_bit_buf_peek(), and then discarded.
152  *
153  * \param[in] bbuf Bit buffer
154  * \return pointer to current internal buffer data
155  */

195 /*! Add bits from a given integer to the buffer.
196  *
197  * Note that the bit region being read is assumed to be justified against the least-significant end of the
198  * integer, though the bits within that region are read from most-significant to least-significant.
199  *
200  * For example, if bits == 0x8F == (10001011)b, and nbits == 4, the bits "1011" will be added to the buffer.
201  *
202  * \param[in,out] bbuf Bit buffer
203  * \param[in] bits Value to draw bits from
204  * \param[in] nbits Number of bits to use (counted from least-significant end, right-to-left)
205  * \param[in] pad Should any bits be added to force the result to end on a byte-boundary
206  */
207 M_API void M_bit_buf_add(M_bit_buf_t *bbuf, M_uint64 bits, size_t nbits, M_bit_buf_pad_t pad);

210 /*! Add bits from a given binary-ascii string to the buffer.
211  *
212  * A binary-ascii string is a list of 1 and 0 characters (e.g., "100010").
213  *
214  * Any whitespace in the string will be silently ignored. So, " 1000 1 0" will add the same data as "100010".
215  *
216  * \param[in,out] bbuf Bit buffer
217  * \param[in] bitstr String to draw bits from
218  * \param[in] pad Should any bits be added to force the result to end on a byte-boundary
219  *
220  * \return M_FALSE on error (given bitstr had characters other than '0', '1', or whitespace).
221  */
222 M_API M_bool M_bit_buf_add_bitstr(M_bit_buf_t *bbuf, const char *bitstr, M_bit_buf_pad_t pad);
