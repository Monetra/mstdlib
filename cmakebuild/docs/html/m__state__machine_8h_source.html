37 /*! \addtogroup m_state_machine State Machine
38  * \ingroup mstdlib_base
39  *
40  * Non-linear state machine for running a sequence of states.
41  *
42  * Normally the state machine will run using a linear or linear hybrid method. When adding states to the state machine
43  * the order is preserved. By default the next state is the next state added. A state can change/set the next state
44  * id. Using the next state is simply a convenience for using the state machine in a linear manner. So a state can
45  * set an id to transition to or it can rely on the next state in the ordered state list to be called next. This
46  * behavior can be disabled and the state machine will run as a pure non-linear state machine where a transition id
47  * is required to be set by a state.
48  *
49  * States are given an integral id. The id must be unique and is used so a state can specify which state to transition
50  * to. No id can use the number zero (0); it is reserved for internal use.
51  *
52  * Each state can have an optional cleanup state machine. The cleanup will be called when the error or done status
53  * are returned by run. All status that ran (including the one that generated the error) will have their cleanup
54  * state machine run.
55  *
56  * Cleanup is run in reverse order that the states were run in. For example states A - E were run and
57  * E returns done. Cleanup for states will run E - A. Due to the state machine supporting non-linear sequences
58  * it is possible that a cleanup machines will be called multiple times.
59  *
60  * Cleanup can be used in multiple ways. They can be resource clean up (particularly useful when the
61  * done_cleanup flag is used) that is run when the machine finishes. Or they can be used as error recovery such as
62  * performing an action if an error occurs (default use).
63  *
64  * Cleanup should be specific to the state and should in some way be based on the state they're associated
65  * with. A final cleanup on success could be handled as a final state but should be handled outside of
66  * the state machine entirely. Such as being handled as part of cleaning up the void pointer of state data.
67  *
68  * Example:
69  *
70  * \code{.c}
71  * typedef enum {
72  * STATE_A = 1,
73  * STATE_B,
74  * STATE_C
75  * } states_t;
76  *
77  * static M_state_machine_status_t state_a(void *data, M_uint64 *next)
78  * {
79  * (void)data;
80  * *next = STATE_C;
81  * return M_STATE_MACHINE_STATUS_NEXT;
82  * }
83  *
84  * static M_state_machine_status_t state_b(void *data, M_uint64 *next)
85  * {
86  * (void)data;
87  * (void)next;
88  * return M_STATE_MACHINE_STATUS_DONE;
89  * }
90  *
91  * static M_state_machine_status_t state_c(void *data, M_uint64 *next)
92  * {
93  * (void)data;
94  * *next = STATE_B;
95  * return M_STATE_MACHINE_STATUS_NEXT;
96  * }
97  *
98  * int main(int argc, char **argv)
99  * {
100  * M_state_machine_t *sm;
101  * M_state_machine_status_t status;
102  *
103  * sm = M_state_machine_create(0, NULL, M_STATE_MACHINE_NONE);
104  *
105  * M_state_machine_insert_state(sm, STATE_A, 0, NULL, state_a, NULL, NULL);
106  * M_state_machine_insert_state(sm, STATE_B, 0, NULL, state_b, NULL, NULL);
107  * M_state_machine_insert_state(sm, STATE_C, 0, NULL, state_c, NULL, NULL);
108  *
109  * do {
110  * status = M_state_machine_run(sm, NULL);
111  * } while (status == M_STATE_MACHINE_STATUS_WAIT);
112  *
113  * if (status != M_STATE_MACHINE_STATUS_DONE) {
114  * M_printf("state machine failure\n");
115  * } else {
116  * M_printf("state machine success\n");
117  * }
118  *
119  * M_state_machine_destroy(sm);
120  * return 0;
121  * }
122  * \endcode
123  *
124  * @{
125  */
126

136 /*! Status of the state machine while running though states. */
137 typedef enum {
138  /* State specific status. Only states can return these. */
139  M_STATE_MACHINE_STATUS_NONE = 0, /*!< Invalid status. */
140  M_STATE_MACHINE_STATUS_NEXT, /*!< Success continue to the next state.
141  The state was run and should be recored as well as cleanup added
142  to the cleanup list. */
143  M_STATE_MACHINE_STATUS_PREV, /*!< A recoverable error occurred. Go to the last successful (non-continue)
144  state.
145 
146  This should be treated as a special case and is primarily a
147  convenience when using the state machine in a linear manner. It should
148  not be used in stead of specifying an id and calling next if it is
149  possible to do so.
150 
151  This does not back out states. State cleanups will not be called when
152  skipping back over states. Also, the list of cleanups will not be
153  modified to remove cleanups for states that have been called. Further,
154  This can result in a state having it's cleanup registered multiple
155  times as a result of multiple successful calls. */
156  M_STATE_MACHINE_STATUS_CONTINUE, /*!< Success continue to the next state.
157  The state was skipped and should be treated as such. The cleanup
158  for this state will not be added to the cleanup list.
159 
160  This should not be treated as next without cleanup. It is for
161  signifying that the state was skipped. If you need next without
162  cleanup the state should be registered without a cleanup state
163  machine. Even if that means having two ids for the same state function
164  one with and one without a cleanup registered. */
165 
166  /* Shared status. States and the state machine can return these. */
167  M_STATE_MACHINE_STATUS_ERROR_STATE, /*!< An unrecoverable error occurred within a state. Exit and clean up.
168  The state is responsible for error reporting though the void data
169  pointer passed to the state function. */
170  M_STATE_MACHINE_STATUS_WAIT, /*!< The state is processing in a non-blocking fashion. More calls to run
171  are required to continue the operation. */
172  M_STATE_MACHINE_STATUS_DONE, /*!< The sequence completed successfully. */
173 
174  /* State machine specific status. Only the state machine can return these. */
175  M_STATE_MACHINE_STATUS_STOP_CLEANUP, /*< Used by cleanup state machines to stop processing further cleanup
176  state machines within a state machine. */
177  /* All of these are unrecoverable errors. */
178  M_STATE_MACHINE_STATUS_ERROR_INVALID, /*!< The state machine was called with an invalid parameter. */
179  M_STATE_MACHINE_STATUS_ERROR_BAD_ID, /*!< Invalid transition specified. Id not found. Most likely the state
180  specified an id to transition to that doesn't exist. */
181  M_STATE_MACHINE_STATUS_ERROR_NO_NEXT, /*!< Invalid transition specified. An next id was not specified. This can
182  happen when running in a linear manner and the last state in the
183  sequence does not return done. There are no states after the last
184  state so we cannot continue with the sequence. */
185  M_STATE_MACHINE_STATUS_ERROR_BAD_NEXT, /*!< Invalid transition specified. The specified next id is not valid (not
186  listed in the states list of next ids) for the state. */
187  M_STATE_MACHINE_STATUS_ERROR_SELF_NEXT, /*!< Invalid transition specified. The specified next id is the current id.
188  Use the continue_loop flag to disable this check. */
189  M_STATE_MACHINE_STATUS_ERROR_NO_PREV, /*!< Invalid transition specified. There are no previous states to
190  transition to. */
191  M_STATE_MACHINE_STATUS_ERROR_INF_CONT /*!< A possible infinite continuation loop has been encountered. */
192 } M_state_machine_status_t;

195 /*! Options to control the behavior of the state machine. */
196 typedef enum {
197  M_STATE_MACHINE_NONE = 0, /*!< Normal operation. */
198  M_STATE_MACHINE_SINGLE_PREV = 1 << 1, /*!< Do not allow multiple states to return STATUS_PREV in a row.
199  Only one PREV return is allowed between NEXT calls. */
200  M_STATE_MACHINE_CONTINUE_LOOP = 1 << 2, /*!< Normally continuations are tracked for the continuation cycle
201  and any continuation that is repeated is treated as an internal
202  error in order to detect and prevent accidental infinite loops.
203 
204  This option disables this check and allows continuations to call
205  continuations that have been called previously. */
206  M_STATE_MACHINE_SELF_CALL = 1 << 3, /*!< Normally states cannot all themselves. This flag also
207  allows states to call themselves. */
208  M_STATE_MACHINE_DONE_CLEANUP = 1 << 4, /*!< State cleanups should be called on done. */
209  M_STATE_MACHINE_ONE_CLEANUP = 1 << 5, /*!< State cleanup should be called once no matter how many times the
210  state was called. */
211  M_STATE_MACHINE_EXPLICIT_NEXT = 1 << 6, /*!< Normally the state machine defaults to using the next state in
212  the order states were added if a state isn't explicits specified
213  by the current state. This requires that a state specify the next
214  (transition) state.
215 
216  This will force the state machine to function purely as a non-linear
217  state machine. The linear / linear hybrid functionality will be
218  disabled. This option cannot be used in conjunction with linear_end.
219  The linear_end flag will be ignored if this flag is set. */
220  M_STATE_MACHINE_LINEAR_END = 1 << 7 /*!< Normally a state machine is done when the done status is returned
221  by a state. This allows the state machine to be considered done if
222  a state does not specify a transition, it returns next or continue
223  and the current state is the last state in the ordered state list. */
224 } M_state_machine_flags_t;

231  M_STATE_MACHINE_CLEANUP_REASON_DONE, /*!< State machine finished successfully. */
232  M_STATE_MACHINE_CLEANUP_REASON_ERROR, /*!< State machine stopped due to error. */

381 /*! Sub state machine post (de-initialization) callback.
382  *
383  * The sub_status argument is the status returned by the sub state machine. Possible status:
384  *
385  * - M_STATE_MACHINE_STATUS_DONE
386  * - M_STATE_MACHINE_STATUS_ERROR_*
387  *
388  * The sub_status will next be M_STATE_MACHINE_STATUS_NEXT or similar. Thus,
389  * the sub_status should not be blindly returned from the post function as it
390  * will stop processing the parent state machine. If processing needs to
391  * continue the sub_status should be checked and M_STATE_MACHINE_STATUS_NEXT or
392  * similar should be returned. M_STATE_MACHINE_STATUS_DONE is the only
393  * successful sub_status that can be set, so patterns that check against
394  * M_STATE_MACHINE_STATUS_DONE should be used. For example:
395  *
396  * \code{c}
397  * if (sub_status == M_STATE_MACHINE_STATUS_DONE) {
398  * // Success and continue.
399  * return M_STATE_MACHINE_STATUS_NEXT;
400  * }
401  * ...
402  * \endcode
403  *
404  * \code{c}
405  * if (sub_status != M_STATE_MACHINE_STATUS_DONE) {
406  * // Error of some kind. Propagate it up.
407  * return sub_status;
408  * }
409  * ...
410  * \endcode
411  *
412  * \code{c}
413  * if (stop_condition) {
414  * // Some kind of external stop condition was encountered.
415  * // Return the sub_status because it will stop processing
416  * // and we should maintain status from the sub state machine.
417  * return sub_status; // Status is error or done.
418  * }
419  * ...
420  * \endcode
421  *
422  * \param[in,out] data An opaque data type storing data that should be passed to the cb.
423  * \param[out] sub_status The status of the last state in the sub state machine.
424  * \param[out] next The next id the state machine should transition to. When operating in linear or a hybrid manner
425  * this will be set to the next linear state. Changing this will change what state is next.
426  *
427  * \return The status.
428  */
429 typedef M_state_machine_status_t (*M_state_machine_post_cb)(void *data, M_state_machine_status_t sub_status, M_uint64 *next);

499 /*! Add a state machine as a state to the state machine.
500  *
501  * The state machine will duplicate the sub state machine and keep a copy.
502  *
503  * The sub state machine will run though all states in the sub state machine. The state machine will return
504  * M_STATE_MACHINE_STATUS_WAIT from the sub state machine and resume the sub state machine when started again.
505  *
506  * The sub state machine's final status will be passed to the post function if one is given. If a post function is
507  * not set, a status of M_STATE_MACHINE_STATUS_DONE will be returned as M_STATE_MACHINE_STATUS_NEXT. This
508  * is to prevent a M_STATE_MACHINE_STATUS_DONE from the sub state machine from accidentally stopping the calling
509  * state machine. If M_STATE_MACHINE_STATUS_DONE is needed as the result of the sub state machine's run then a
510  * post function is necessary.
511  *
512  * \param[in,out] m The state machine.
513  * \param[in] id The id associated with this state. Must be unique.
514  * \param[in] ndescr A numeric description of the state. Can be 0.
515  * \param[in] descr A textual description of the state. Can be NULL.
516  * \param[in] subm The state machine that should be called from this one. Cannot be NULL.
517  * \param[in] pre A function to call before the sub state machine is started. Can be NULL.
518  * \param[in] post A function to call after the sub state machine is finished. Can be NULL.
519  * \param[in] cleanup The cleanup state machine to call. Can be NULL if no cleanup is necessary for this state.
520  * \param[in] next_ids A list of valid transitions for this state. Can be NULL to denote all states are
521  * valid transitions. If not NULL the state machine takes ownership of next_ids.
522  *
523  * \return M_TRUE if the sub state machine was added. Otherwise M_FALSE.
524  */
525 M_API M_bool M_state_machine_insert_sub_state_machine(M_state_machine_t *m, M_uint64 id, M_uint64 ndescr, const char *descr, const M_state_machine_t *subm, M_state_machine_pre_cb pre, M_state_machine_post_cb post, M_state_machine_cleanup_t *cleanup, M_list_u64_t *next_ids);

684 /*! Run the state machine.
685  *
686  * This may need to be called multiple times. A state can run non-blocking (poll based) where
687  * the state can return a wait state. The wait state means it finished processing but has
688  * more to do.
689  *
690  * On error the cleanup state machine for the state will be called. When returning from a sub state machine
691  * which had clean up run the post function which can override and ignore an error can stop the cleanup
692  * process. Thus cleanup can be stopped and the state machine can recover from the error that started
693  * the process.
694  *
695  * \param[in,out] m The state machine to run.
696  * \param[in,out] data State specific data that can be used and or manipulated by each state.
697  *
698  * \return Result. The done and wait returns are the only successful results (wait requiring additional calls).
699  * All other results are error conditions.
700  */

706 /*! Rest a running state machine.
707  *
708  * A condition outside of the state machine could determine it needs to restart while it was
709  * in a running state. Not specifically running but in the middle of a run; having returned from
710  * a wait state for example. This will reset the state machine's internal process state so that
711  * it can be started from the beginning again.
712  *
713  * This will not run cleanup immediately if requested but instead sets the state machine to start
714  * cleanup on next run. The sub state machine post function will not allow overriding the cleanup
715  * result and prevents the state machine from stopping cleanup. M_state_machine_run *MUST* be called.
716  * Also, remember that cleanup state machines can call wait so it may be necessary to run multiple times.
717  *
718  * \param[in,out] m The state machine.
719  * \param[in] reason Whether state cleanups should run. Cleanup callbacks told cleanup is due to the
720  * reason code. Use M_STATE_MACHINE_CLEANUP_NONE to prevent cleanup.
