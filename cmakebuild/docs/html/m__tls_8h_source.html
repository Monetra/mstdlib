60 /*! Certificate verification level.
61  *
62  * Used by client connections to control how they decide to trust
63  * the certificate presented by the server.
64  *
65  */
66 typedef enum {
67  M_TLS_VERIFY_NONE = 0, /*!< Do not verify the certificate or hostname. */
68  M_TLS_VERIFY_CERT_ONLY = 1, /*!< Only verify the certificate. The domain name is not checked. */
69  M_TLS_VERIFY_CERT_FUZZY = 2, /*!< Verify the certificate and that the base domain name matches.
70  Use this for servers that don't properly have a wild card cert
71  but still use a sub domain. */
72  M_TLS_VERIFY_FULL = 3 /*!< Default. Verify the certificate and full domain name matches */
73  /* XXX: OCSP, CRL? */
74 } M_tls_verify_level_t;

153 /* Set a certificate that will be sent to the server for the server to validate.
154  *
155  * Keys must be RSA or DSA.
156  * Certificates must be X509 and can be PEM, or DER encoded.
157  *
158  * The intermediate file can have multiple certificates comprising a certificate chain.
159  * Certificates in the chain should be ordered by signing with the certificate that signed
160  * the client certificate on top. The root CA file should not be in this chain.
161  *
162  * Intermediate example:
163  *
164  * crt (not part of intermediate file)
165  * Intermediate 1 that signed crt
166  * Intermediate 2 that signed intermediate 1
167  * Intermediate 3 that signed intermediate 2
168  * CA that signed intermediate 3 (not part of intermediate file).
169  *
170  * \param[in] ctx Client context.
171  * \param[in] key Private key associated with certificate.
172  * \param[in] key_len Length of private key.
173  * \param[in] crt Certificate.
174  * \param[in] crt_len Length of certificate.
175  * \param[in] intermediate Intermediate certificate chain.
176  * \param[in] intermediate_len Length of intermediate certificate chain.
177  *
178  * \return M_TRUE on success, otherwise M_FALSE on error.
179  *
180  * \see M_tls_clientctx_set_cert_files
181  */
182 M_API M_bool M_tls_clientctx_set_cert(M_tls_clientctx_t *ctx, const unsigned char *key, size_t key_len, const unsigned char *crt, size_t crt_len, const unsigned char *intermediate, size_t intermediate_len);
