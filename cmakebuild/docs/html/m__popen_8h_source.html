37 /*! \addtogroup m_popen Process Open (popen)
38  * \ingroup m_thread
39  *
40  * Open and interact with a process.
41  *
42  * Example:
43  *
44  * \code{.c}
45  * const char *data = "<x><t>data</t></x>"
46  * M_popen_err_t mperr;
47  * M_popen_handle_t mp;
48  * M_popen_status_t status;
49  * int retval;
50  * char *stdout_buf = NULL;
51  * size_t stdout_buf_len = 0;
52  * char *stderr_buf = NULL;
53  * size_t stderr_buf_len = 0;
54  *
55  * mp = M_popen("curl <url>", &mperr);
56  * if (mp == NULL) {
57  * printf("m_popen failed: %s\n", M_popen_strerror(mperr));
58  * return M_FALSE;
59  * }
60  *
61  * M_printf("Process spawned....\n");
62  *
63  * retval = M_popen_write(mp, M_POPEN_FD_WRITE, data, M_str_len(data));
64  * if (retval <= 0) {
65  * M_printf("M_popen_write failed, retval = %d\n", retval);
66  * M_popen_close(mp, &mperr);
67  * return M_FALSE;
68  * }
69  *
70  * / * Close file descriptor to let process know we're done * /
71  * if (!M_popen_closefd(mp, M_POPEN_FD_WRITE)) {
72  * M_printf("M_popen_closefd() failed\n");
73  * M_popen_close(mp, &mperr);
74  * return M_FALSE;
75  * }
76  *
77  * M_printf("Wrote process stream....\n");
78  *
79  * while ((status=M_popen_check(mp)) == M_POPEN_STATUS_RUNNING) {
80  * M_thread_sleep(50000);
81  * }
82  *
83  * if (status == M_POPEN_STATUS_ERROR) {
84  * retval = M_popen_close(mp, &mperr);
85  * printf("Error during M_popen_check(): %d: %s\n", retval, M_popen_strerror(mperr));
86  * return M_FALSE;
87  * }
88  *
89  * M_printf("Process done...\n");
90  *
91  * retval = M_popen_close_ex(mp, &stdout_buf, &stdout_buf_len, &stderr_buf, &stderr_buf_len, &mperr, 0);
92  * if (retval < 0) {
93  * M_printf("error: %s\n", M_popen_strerror(mperr));
94  * return M_FALSE;
95  * }
96  *
97  * M_printf("stdout: %d:\n%s\n", (int)stdout_buf_len, stdout_buf);
98  * M_printf("stderr: %d:\n%s\n", (int)stderr_buf_len, stderr_buf);
99  * M_free(stdout_buf);
100  * M_free(stderr_buf);
101  * M_printf("return code: %d\n", retval);
102  * \endcode
103  *
104  * @{
105  */

142 /*! Start the specified command and open stdin (write), stdout (read), and
143  * stderr (read) file descriptors for communication.
144  *
145  * Must call M_popen_close() to clean up the returned handle.
146  *
147  * \param[in] cmd Command to execute.
148  * \param[out] errorid Pointer to store error id if an error occurs.
149  *
150  * \return NULL on failure, M_popen_handle_t on success.
151  */

155 /*! Read from a file descriptor
156  *
157  * \param[in] mp Open M_popen_t object.
158  * \param[in] fd Which FD to read from.
159  * \param[out] out Buffer to hold read data.
160  * \param[in] out_len Length of out buffer.
161  * \param[in] timeout_ms Time in ms to wait for data.

209 /*! Close the M_popen_t object.
210  *
211  * This will perform a blocking wait for the process to exit before returning
212  * control to the caller.
213  *
214  * \param[in] mp M_popen_t object
215  * \param[out] stdout_buf Optional parameter. Will return allocated buffer containing
216  * the contents of the process's stdout. If specified, must
217  * also specify stdout_buf_len.
218  * \param[out] stdout_buf_len Optional parameter. Will return the length of stdout_buf.
219  * \param[out] stderr_buf Optional parameter. Will return allocated buffer containing
220  * the contents of the process's stderr. If specified, must
221  * also specify stderr_buf_len.
222  * \param[out] stderr_buf_len Optional parameter. Will return the length of stderr_buf.
223  * \param[out] errorid if an error has occurred, will populate with a
224  * reason code.
225  * \param[in] timeout Time in ms to wait for the processes to exit. If the process
226  * has not finished after the timeout expires it will be killed.
227  * M_TIMEOUT_INF will cause this to block until the process exits.
228  * Note: the time out only has 15 ms resolution.
229  *
230  * \return -1 on error, -2 on timeout, otherwise the exit code from the process.
231  */
232 M_API int M_popen_close_ex(M_popen_handle_t *mp, char **stdout_buf, size_t *stdout_buf_len, char **stderr_buf, size_t *stderr_buf_len, M_popen_err_t *errorid, M_uint64 timeout);

235 /*! Close the M_popen_t object.
236  *
237  * This is a simplified wrapper around M_popen_close_ex(). This command blocks forever until the
238  * child process is done. If you need to force-kill the process after a given timeout, use
239  * M_popen_close_ex() instead of this function.
240  *
241  * \param[in] mp M_popen_t object
242  * \param[out] errorid if an error has occurred, will populate with a reason code.
243  *
244  * \see M_popen_close_ex
245  *
246  * \return -1 on error, otherwise the exit code from the process.
247  */
248 M_API int M_popen_close(M_popen_handle_t *mp, M_popen_err_t *errorid);
