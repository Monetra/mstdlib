55 /*! Verify that the given single email address is valid.
56  *
57  * An email address is composed of the following parts:
58  * \verbatim display_name<local_part@domain>\endverbatim
59  *
60  * The display name may contain double-quotes. Within double-quotes, the full range of symbols may be used (dot,
61  * comma, semicolon, at, escaped double-quote, etc.). The space and tab characters are allowed in the display name,
62  * even outside of quotes.
63  *
64  * If the display name is not provided, the angle brackets around the rest of the email address are optional.
65  *
66  * This email address validator differs from the RFC standards for email in the following ways:
67  *
68  * \li Double quotes are not allowed, except in the display name (a.k.a. friendly name).
69  * \li Comments (chunks of text enclosed by parentheses) are not allowed, though parenthesis are allowed in the
70  * display name if they're surrounded by double-quotes.
71  * \li Consecutive dots are allowed in the local part of the address, since they are accepted by gmail.
72  * \li Only the following non-alphanumeric characters are allowed in the local part: \verbatim '*+-=^_{}.~ \endverbatim
73  * \li IP address domains are not allowed (e.g., joe.smith\@[192.168.2.1] is invalid).
74  * \li UTF-8 is not allowed in any part of the address (display name, local part, or domain name).
75  *
76  * Example of valid address:
77  * \verbatim "Public, John Q. (Support)" <jq..public+support@generic-server.com> \endverbatim
78  *
79  * \see M_verify_email_address_list
80  *
81  * \param[in] addr email address to verify
82  * \return TRUE if the address is valid
83  */
84 M_API M_bool M_verify_email_address(const char *addr);
85

87 /*! Verify that the given domain name is valid.
88  *
89  * Valid domain names are composed of letters, digits, and hyphens ('-'), separated into sections
90  * by dots ('.'). Each section is limited to 63 chars, and may not begin or end with a hyphen.
91  * The entire domain name is limited to 253 characters - note that a 253 character domain name
92  * takes 255 bytes to store with the length-prefixed encoding used by DNS.
93  *
94  * \param[in] dname domain name to verify
95  * \return TRUE if the domain name is valid
96  */
97 M_API M_bool M_verify_domain(const char *dname);

100 /*! Verify that the given list of email addresses is valid.
101  *
102  * The list may be delimited by either commas (',') or semicolons (';'), but not both.
103  * This function limits the number of addresses in the list to 100 - this isn't a true
104  * standard, but it's the limit used virutally everywhere in practice.
105  *
106  * The list parser is tolerant of whitespace - individual addresses are trimmed before they're
107  * passed to the email address validator.
108  *
109  * The individual addresses in the list may include display names (a.k.a friendly names).
110  *
111  * \see M_verify_email_address
112  *
113  * \param[in] addr_list list of email addresses to verify
114  * \param[in] delim_type Type of delimiter to expect. Can be auto detected.
115  * \return TRUE if the list and all emails in it are valid
116  */
117 M_API M_bool M_verify_email_address_list(const char *addr_list, M_verify_email_listdelim_t delim_type);
