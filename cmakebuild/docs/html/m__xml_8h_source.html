39 /*! \addtogroup m_xml XML
40  * \ingroup m_formats
41  *
42  * DOM based XML data processing.
43  *
44  * This is a simple API for reading, creating, manipulating, searching, and writing
45  * XML data.
46  *
47  * It is possible to construct an invalid XML document. It is also possible that
48  * an invalid XML document created with M_xml will not be readable by M_xml.
49  * Specifically, when dealing with attribute and text encoding options for M_xml_read
50  * and M_xml_write. Also, attribute keys and tag names are not validated to ensure
51  * they are valid XML names. This must happen outside of M_xml.
52  *
53  * M_xml is not susceptible to common XML entity expansion attaches such as
54  * billion laughs, quadratic blowup, and external entity expansion. Only basic
55  * XML entities are expanded and only one level is expanded. Further external
56  * entity references are not downloaded. Neither are external DTDs. This will
57  * mitigate against network based attacks relying on retrieval.
58  *
59  * Example:
60  *
61  * \code{.c}
62  * M_xml_node_t *xml;
63  * const char *data = "<r><tag1>abc</tag1><tag2>123</tag2><tag3>xyz</tag3></r>";
64  * char *out;
65  *
66  * xml = M_xml_read(data, M_str_len(data), M_XML_READER_NONE, NULL, NULL, NULL);
67  * if (xml == NULL) {
68  * M_printf("xml parse error\n");
69  * return M_FALSE;
70  * }
71  *
72  * M_xml_create_element_with_text("AbCd", "blah blah blah", 0, xml);
73  *
74  * out = M_xml_write(xml, M_XML_WRITER_LOWER_TAGS|M_XML_WRITER_LOWER_ATTRS|M_XML_WRITER_PRETTYPRINT_SPACE, NULL);
75  * M_printf("out=\n%s\n", out);
76  * M_free(out);
77  * \endcode
78  *
79  * @{
80  */

86 /*! Node types in our XML tree structure */
87 typedef enum {
88  M_XML_NODE_TYPE_UNKNOWN = 0, /*!< An invalid node type. */
89  M_XML_NODE_TYPE_DOC = 1, /*!< The base of every XML tree and provides access to the
90  documents data.
91  Supports: Nodes.
92  Does not support: Name, Attributes, Text, Tag data. */
93  M_XML_NODE_TYPE_ELEMENT = 2, /*!< Element (tag).
94  E.g: \code{xml}<name key="val" />, <name>...</name>\endcode.
95  Supports: Attributes, Nodes, Name.
96  Does not support: Text, Tag data. */
97  M_XML_NODE_TYPE_TEXT = 3, /*!< Text.
98  E.g: abc.
99  Supports: Text.
100  Does not support: Nodes, Name, Attributes, Tag data. */
101  M_XML_NODE_TYPE_PROCESSING_INSTRUCTION = 4, /*!< Conveys information.
102  E.g: \code{xml}<?xml version="1.0" encoding="UTF-8" ?>\endcode
103  Supports: Name, Attributes, Tag data.
104  Does not support: Nodes, Text. */
105  M_XML_NODE_TYPE_DECLARATION = 5, /*!< HTML/DTD declaration.
106  E.g: \code{xml}<!DOCTYPE html>, <!ELEMENT element-name category>, <!ATTLIST payment type CDATA "check">\endcode
107  Supports: Name, Tag data.
108  Does not support: Nodes, Attributes, text. */
109  M_XML_NODE_TYPE_COMMENT = 6 /*!< Comment.
110  E.g: \code{xml}<!-- comment -->\endcode
111  Supports: Tag data.
112  Does not support: Nodes, Name, Attributes, Text. */
113 } M_xml_node_type_t;

116 /*! Flags to control the behavior of the XML writer. */
117 typedef enum {
118  M_XML_READER_NONE = 0, /*!< Normal operation. */
119  M_XML_READER_IGNORE_COMMENTS = 1 << 0, /*!< Don't include comments as part of the output during parse. */
120  M_XML_READER_TAG_CASECMP = 1 << 1, /*!< Compare open and close tags case insensitive. */
121  M_XML_READER_DONT_DECODE_TEXT = 1 << 2, /*!< By default text data will be xml entity decoded.
122  This prevents the decode from taking place. It should be paired
123  with the equivalent don't encode option when writing. */
124  M_XML_READER_DONT_DECODE_ATTRS = 1 << 3 /*!< By default attribute values will be attribute entity decoded.
125  This prevents the decode from taking place. It should be paired
126  with the equivalent don't encode option when writing. */
127 } M_xml_reader_flags_t;

130 /*! Flags to control the behavior of the XML reader. */
131 typedef enum {
132  M_XML_WRITER_NONE = 0, /*!< No indent. All data on a single line. */
133  M_XML_WRITER_IGNORE_COMMENTS = 1 << 0, /*!< Comments are not included. */
134  M_XML_WRITER_LOWER_TAGS = 1 << 1, /*!< All tags are written lower case. */
135  M_XML_WRITER_LOWER_ATTRS = 1 << 2, /*!< All attribute keys are written lower case. */
136  M_XML_WRITER_DONT_ENCODE_TEXT = 1 << 3, /*!< By default text data will be xml entity encoded.
137  This prevents the encode from taking place. It should be paired
138  with the equivalent don't decode option when reading. */
139  M_XML_WRITER_DONT_ENCODE_ATTRS = 1 << 4, /*!< By default attribute values will be attribute entity encoded.
140  This prevents the encode from taking place. It should be paired
141  with the equivalent don't decode option when reading. */
142  M_XML_WRITER_PRETTYPRINT_SPACE = 1 << 5, /*!< 2 space indent. */
143  M_XML_WRITER_PRETTYPRINT_TAB = 1 << 6, /*!< Tab indent. */
144  M_XML_WRITER_SELFCLOSE_SPACE = 1 << 7 /*!< Add a space before the closing slash for self closing tags. */
145 } M_xml_writer_flags_t;

389 /*! Using XPath expressions, scan for matches.
390  *
391  * Note that full XPath support does not yet exist. Also only element nodes are currently returned unless text() is
392  * used to return text nodes.
393  *
394  * Supports:
395  * Syntax | Meaning
396  * -------------------|---------
397  * tag | Selects children of the current element node with the given tag name.
398  * *:tag | Selects children of the current element node with the given tag name ignoring any namespace on the tag. Also matches tags without a namespace.
399  * * | Selects all children of the current element node.
400  * / | Selects children one level below the current element node.
401  * // | Selects children on all levels bellow the current element node.
402  * . | Selects the current element node.
403  * .. | Selects the parent of the current element node.
404  * text() | Selects all text nodes.
405  * [\@attrib] | Selects elements which have an attribute attrib.
406  * [\@attrib=val] | Selects elements which have an sttribute attrib with a value of val.
407  * [\@attrib="val"] | Selects elements which have an sttribute attrib with a value of val.
408  * [\@attrib='val'] | Selects elements which have an sttribute attrib with a value of val.
409  * [\@*] | Selects elements which have an (any) attribute set.
410  * [idx] | Select an element at a given position.
411  * [position() ? idx] | Select an element at a given position.
412  *
413  * More information about [idx]:
414  *
415  * Index matching can either be an integer (starting from 1) or the expression "last()". "last()" can
416  * be followed by a negative integer in order to count backwards from the last elements. For example: "last()-1"
417  * is the second to last element. The idx is not the index within the parent the node but the index of the
418  * matching nodes. idx must be preceded by either a tag or text().
419  *
420  * More information about [position() ? idx]:
421  *
422  * Position matching can use the equality modifiers '=', '<=', '>=', '<', '>'. These will select one or
423  * more nodes that match the given criteria. "last()" can be used as the index.
424  *
425  * E.g: "a/b[2]" for `<a><b/><c/><b/></a>` will select the second "b" after "c".
426  *
427  * \param[in] node The node.
428  * \param[in] search search expression
429  * \param[in] flags M_xml_reader_flags_t flags to control the behavior of the search.
430  * valid flags are:
431  * - M_XML_READER_NONE
432  * - M_XML_READER_TAG_CASECMP
433  * \param[out] num_matches Number of matches found
434  *
435  * \return array of M_xml_node_t pointers on success (must free array, but not internal pointers), NULL on failure
436  */
437 M_API M_xml_node_t **M_xml_xpath(M_xml_node_t *node, const char *search, M_uint32 flags, size_t *num_matches) M_MALLOC;
