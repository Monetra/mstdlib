37 /*! \addtogroup m_ini INI
38  * \ingroup m_formats
39  *
40  * Configurable handling for various formats. Such as # vs ; comment identifiers.
41  *
42  * For easier access functions that do not take a section use the key form
43  * 'section/key'. If multiple '/' characters are in the combined key the section
44  * is only up until the first '/'. Meaning: 'section/key/key_part'
45  *
46  * Can handle multiple or single values under a single key.
47  *
48  * Supports:
49  * - Read
50  * - Write
51  * - Modify
52  * - Merge
53  *
54  * Example:
55  *
56  * \code{.c}
57  * M_ini_t *ini = NULL;
58  * M_ini_settings_t *info = NULL;
59  * char *out;
60  * size_t errln = 0;
61  *
62  * info = M_ini_settings_create();
63  * M_ini_settings_set_quote_char(info, '"');
64  * M_ini_settings_set_escape_char(info, '"');
65  * M_ini_settings_set_padding(info, M_INI_PADDING_AFTER_COMMENT_CHAR);
66  * M_ini_settings_reader_set_dupkvs_handling(info, M_INI_DUPKVS_REMOVE);
67  * M_ini_settings_writer_set_multivals_handling(info, M_INI_MULTIVALS_USE_LAST);
68  *
69  * ini = M_ini_read_file("file.ini", info, M_TRUE, &errln, 0);
70  * if (ini == NULL) {
71  * M_printf("ini could not be parsed. Error line: %zu\n", errln);
72  * return M_FALSE;
73  * }
74  *
75  * M_ini_kv_set(ini, "s1/key1", "yes");
76  *
77  * out = M_ini_write(ini, info);
78  * M_printf("new ini=\n%s\n", out);
79  * M_free(out);
80  *
81  * M_ini_destroy(ini);
82  * M_ini_settings_destroy(info);
83  * \endcode
84  *
85  * @{
86  */

97 /*! Conflict handler function prototype.
98  *
99  * Ini merging can have conflicts resolved using a callback function. The use of the
100  * resolution callback is dependent on the appropriate merge flag being set.
101  *
102  * \param[in] key The key. If key is NULL then the values are they key. In this case if the value is NULL then
103  * the key doesn't exist for that location.
104  * \param[in] val_cur The value in the current ini.
105  * \param[in] val_new The new value.
106  *
107  * \return M_TRUE if the current value should be used. Otherwise, M_FALSE if the new value should be used.
108  */
109 typedef M_bool (*M_ini_merge_resolver_t)(const char *key, const char *val_cur, const char *val_new);

114 /*! Duplicate key, value pair handling where a key is encountered multiple times. */
115 typedef enum {
116  M_INI_DUPKVS_COMMENT_PREV = 0, /*!< Turn previous kv into comments. Last wins.*/
117  M_INI_DUPKVS_REMOVE_PREV, /*!< Remove previous kv from the tree. Last wins. */
118  M_INI_DUPKVS_COMMENT, /*!< Turn the current kv into a comment. First wins. */
119  M_INI_DUPKVS_REMOVE, /*!< Remove the current kv from the tree. First wins. */
120  M_INI_DUPKVS_COLLECT /*!< Multiple kv are allowed and their values should be collected. All win. */
121 } M_ini_dupkvs_t;

124 /*! Control padding when between parts of elements.
125  * Primarily used for writing but also used for reading when a comment duplicate key flag is used. */
126 typedef enum {
127  M_INI_PADDING_NONE = 0, /*!< No padding. */
128  M_INI_PADDING_BEFORE_KV_DELIM = 1 << 0, /*!< Put a space before the kv delimiter. */
129  M_INI_PADDING_AFTER_KV_DELIM = 1 << 1, /*!< Put a space after the kv delimiter. */
130  M_INI_PADDING_AFTER_KV_VAL = 1 << 2, /*!< Put a space after the kv val if followed by a comment. */
131  M_INI_PADDING_AFTER_COMMENT_CHAR = 1 << 3 /*!< Put a space after the comment character. */
132 } M_ini_padding_t;

135 /*! Control how muli value keys are written. */
136 typedef enum {
137  M_INI_MULTIVALS_USE_LAST = 0, /*!< Multi-value keys are not supported. Use the last value. */
138  M_INI_MULTIVALS_USE_FIRST, /*!< Multi-value keys are not supported. Use the first value. */
139  M_INI_MULTIVALS_KEEP_EXISTING, /*!< Multi-value keys are supported. Keep existing values in the same location
140  and place new values after. */
141  M_INI_MULTIVALS_MAINTAIN_ORDER /*!< Multi-value keys are supported. Remove all existing keys and write them all
142  together maintaining the current value order. */
143 } M_ini_multivals_t;

146 /*! Control how conflicts are handled during merge.
147  *
148  * These values all override the default behavior.
149  * Default behavior:
150  * - When a key is in new but not in cur and orig remove the key.
151  * - When the value (single) of cur is the same as orig but different than new use the new value.
152  * - When a key with multiple values has a value that is in cur and orig but not in new remove the value.
153  */
154 typedef enum {
155  M_INI_MERGE_CALLBACK_FUNC = 0, /*!< Use a conflict resolution callback function to determine how to
156  handle conflicts. A callback function must be set otherwise
157  the default handling will be used. */
158  M_INI_MERGE_NEW_REMOVED_KEEP = 1 << 0, /*!< When a key is not in new but in cur and orig keep the key.
159  The default is to remove the key. */
160  M_INI_MERGE_NEW_CHANGED_USE_CUR = 1 << 1, /*!< When the value of cur is the same as orig but different than
161  new use the value from cur. Meaning the default value is set
162  but has changed. The default is to use the new value. */
163  M_INI_MERGE_MULTI_NEW_REMOVED_KEEP = 1 << 3 /*!< When a key with multiple values has a value that is in cur
164  and orig but not in new keep the value. The default is to remove
165  the value. */
166 } M_ini_merge_conflict_t;

679 /*! Merge a new ini into an existing ini.
680  *
681  * The merge processes is similar to a three way diff. The current values are compared to the values
682  * in new and the original. The merge process is:
683  *
684  * 1. Update keys.
685  * a. Only in new = in merged.
686  * b. Only in cur = in merged.
687  * c. In cur and new but not in orig = in merged.
688  * d. In orig and cur but not in new = flag handling (default: not in merged).
689  * e. In orig and new but not in cur = flag handling (default: not in merged).
690  * f. in cur, new and orig = in merged.
691  * 2. Update vals.
692  * a. Cur and orig the same but new different = flag handling (default: use new).
693  * b. Cur and new the same but orig different = use cur/new.
694  * c. New and orig the same but cur different = use cur
695  * d. All there are the same = use cur/new/orig.
696  * 3. Update multi-vals.
697  * a. In cur and new = use cur/new.
698  * b. Only in cur = use cur.
699  * c. In cur and orig but not in new = flag (default remove).
700  * d. In new but not in cur or orig = use new.
701  *
702  * \param[in] cur_ini The current ini. Contains user changes that differ from the original ini.
703  * \param[in] new_ini The new ini.
704  * \param[in] orig_ini The original ini that cur_ini is based on.
705  * \param[in] info Settings controlling how the ini should be merged.
706  */
707 M_API M_ini_t *M_ini_merge(const M_ini_t *cur_ini, const M_ini_t *new_ini, const M_ini_t *orig_ini, const M_ini_settings_t *info);
