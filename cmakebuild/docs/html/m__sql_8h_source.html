68  * MySQL/MariaDB Driver
69  *
70  * Driver Name: mysql
71  *
72  * Driver Connection String Options:
73  * - db: Required. Database Name.
74  * - socketpath: Conditional. If using Unix Domain Sockets to connect to MySQL,
75  * this is the path to the Unix Domain Socket. Use the keyword of 'search'
76  * to search for the socket based on standard known paths. Cannot be used
77  * with host.
78  * - host: Conditional. If using IP or SSL/TLS to connect to MySQL, this is the

86  * this at the default.
87  * - mysql_charset: Optional. Used during table creation, defaults to UTF8.
88  * - max_isolation: Optional. Sets the maximum isolation level used for transactions.
89  * This is used to overwrite requests for SERIALIZABLE isolation levels,
90  * useful with Galera-based clusters that do not truly support Serializable

317 /*! Flags controlling behavior of the connection pool */
318 typedef enum {
319  M_SQL_CONNPOOL_FLAG_NONE = 0, /*!< No special pool flags */
320  M_SQL_CONNPOOL_FLAG_PRESPAWN_ALL = 1 << 0, /*!< Pre-spawn all connections, not just the first.
321  * Without this, the remaining connections are on-demand */
322  M_SQL_CONNPOOL_FLAG_NO_AUTORETRY_QUERY = 1 << 1, /*!< If a non-transactional query is rolled back due to a deadlock
323  * or connectivity failure, the default behavior is to automatically
324  * retry the query, indefinitely. For queries executed as part of
325  * a transaction, rollbacks must be handled by the caller as they
326  * may be dependent on prior queries in the transaction. This flag
327  * will turn off the auto-retry logic */
328  M_SQL_CONNPOOL_FLAG_LOAD_BALANCE = 1 << 2, /*!< If there are multiple servers specified for the connection string,
329  * this will load balance requests across the servers instead of using
330  * them for failover. */

334 /*! Create an SQL connection pool.
335  *
336  * A connection pool is required to be able to run SQL transactions. An internal
337  * connection is automatically claimed for a transaction or statement, or will
338  * wait on an available connection.
339  *
340  * \note The pool is not started untile M_sql_connpool_start() is called, which must
341  * occur before the pool can be used by M_sql_stmt_execute() or M_sql_trans_begin().
342  *
343  * \warning Pool modifications such as M_sql_connpool_add_readonly_pool() and M_sql_connpool_add_trace()
344  * must be called prior to M_sql_connpool_start().
345  *
346  * \param[out] pool Newly initialized pool object
347  * \param[in] driver Name of driver to use. If the driver is not already loaded, will attempt
348  * to load the driver module automatically. Driver modules are named
349  * mstdlib_sql_$driver.dll or mstdlib_sql_$driver.so as appropriate.
350  * \param[in] conn_str A driver-specific connection string or DSN. This string often configures
351  * the host/port, and available options for the driver in use. The connection
352  * strings are a set of key/value pairs, with keys seperated from the values
353  * with an equal sign (=), and values separated by a semi-colon (;). If quoting
354  * is in use, a single-quote (') is recognized, and an escape character of a
355  * single quote (') can be used such that to use a real single quote, you would
356  * use two single quotes. E.g. :
357  * \code host=10.130.40.5:3306;ssl=yes \endcode
358  * Please see the documentation for your driver for available configuration options.
359  * \param[in] username Connection username.
360  * \param[in] password Connection password.
361  * \param[in] max_conns Maximum number of SQL connections to attempt to create. Valid range 1-1000.
362  * \param[in] flags Bitmap of #M_sql_connpool_flags_t options to configure.
363  * \param[out] error Buffer to hold error message.
364  * \param[in] error_size Size of error buffer passed in.
365  * \return #M_SQL_ERROR_SUCCESS on successful pool creation, otherwise one of the #M_sql_error_t errors.
366  */
367 M_API M_sql_error_t M_sql_connpool_create(M_sql_connpool_t **pool, const char *driver, const char *conn_str, const char *username, const char *password, size_t max_conns, M_uint32 flags, char *error, size_t error_size);

370 /*! Create a read-only pool attached to our already-created pool.
371  *
372  * The read-only pool will automatically route SELECT transactions, not part of a
373  * transaction (e.g. not within a M_sql_trans_begin() ... M_sql_trans_commit() block)
374  * to the read-only pool. This can be useful for report generation, where the data
375  * is coming from an asyncronous replication pool for reducing load on the master.
376  *
377  * \note The caller can optionally use M_sql_stmt_set_master_only() to enforce routing of
378  * SELECT transactions to the read/write pool instead.
379  *
380  * The read-only pool must share the same driver, username, password, and usage flags as
381  * specified via M_sql_connpool_create(), and must be called before M_sql_connpool_start().
382  *
383  * Only a single read-only pool per pool object is allowed, repeated calls to this function
384  * will result in a failure.
385  *
386  * \param[in] pool Initialized pool object by M_sql_connpool_create().
387  * \param[in] conn_str A driver-specific connection string or DSN. This string often configures
388  * the host/port, and available options for the driver in use. The connection
389  * strings are a set of key/value pairs, with keys seperated from the values
390  * with an equal sign (=), and values separated by a semi-colon (;). If quoting
391  * is in use, a single-quote (') is recognized, and an escape character of a
392  * single quote (') can be used such that to use a real single quote, you would
393  * use two single quotes. E.g. :
394  * \code host=10.130.40.5:3306;ssl=yes \endcode
395  * Please see the documentation for your driver for available configuration options.
396  * \param[in] max_conns Maximum number of SQL connections to attempt to create. Valid range 1-1000.
397  * \param[out] error Buffer to hold error message.
398  * \param[in] error_size Size of error buffer passed in.
399  * \return #M_SQL_ERROR_SUCCESS on successful readonly pool creation, otherwise one of the #M_sql_error_t errors.
400  */
401 M_API M_sql_error_t M_sql_connpool_add_readonly_pool(M_sql_connpool_t *pool, const char *conn_str, size_t max_conns, char *error, size_t error_size);

404 /*! Set timeouts for connections on the pool. Timeouts can be used to prevent
405  * stale connections from being used if known firewall timers expire, or to
406  * force reconnects to possibly rebalance connections across multiple servers.
407  *
408  * Typically these should be set before M_sql_connpool_start() however it is
409  * safe to change these on an active pool.
410  *
411  * \param[in] pool Initialized connection pool object
412  * \param[in] reconnect_time_s How many seconds to allow a connection to be used before a disconnection
413  * is forced. The connection will be terminated even if not idle, termination
414  * will occur when a connection is returned to the pool instead of prior to use
415  * to prevent unexpected delays. This can be used to either redistribute load
416  * after a node failure when load balancing, or to fall back to a prior host.
417  * Set to 0 for infinite, set to -1 to not change the current value. Default is 0.
418  * \param[in] max_idle_time_s Maximum amount of time a connection can have been idle to be used. Some
419  * firewalls may lose connection state after a given duration, so it may be
420  * advisable to set this to below that threshold so the connection will be
421  * forcibly terminated rather than use. The connection will be terminated
422  * before use and the consumer will attempt to grab a different connection
423  * from the pool, or start a new one if none are available. Set to 0 for infinite,
424  * set to -1 to not change the current value. Default is 0.
425  * \param[in] fallback_s Number of seconds when a connection error occurs to a host before it is
426  * eligible for "fallback". If this isn't set, the only time the first host
427  * will be re-used is if the secondary host(s) also fail. This should be used
428  * in conjunction with reconnect_time_s. Set to 0 to never fallback, or -1 to
429  * not change the current value. Not relevant for load balancing, the host will
430  * always be in the attempt pool. Default is 0.
431  */
432 M_API void M_sql_connpool_set_timeouts(M_sql_connpool_t *pool, M_time_t reconnect_time_s, M_time_t max_idle_time_s, M_time_t fallback_s);

516 /*! Generate a time-based + random unique id suitable for primary key use rather
517  * than using an auto-increment column.
518  *
519  * It is not recommended to use auto-increment columns for portability reasons,
520  * therefore a suitable unique id needs to be chosen that has a low probability
521  * of conflict. There is no guarantee this key is unique, so integrators
522  * should handle conflicts by regenerating the key and re-attempting the
523  * operation (even though this may be an extremely low probability).
524  *
525  * The generated key is a combination of the current timestamp in UTC and
526  * a random suffix. The reason for a timestamp prefix is some databases
527  * cannot handle purely random numbers as they cause index splits that cause
528  * exponential slowdown as the number of rows increase. This can be observed
529  * with MySQL in particular.
530  *
531  * Where possible, a 64bit (signed) column should be used for the unique id,
532  * which has a maximum length of 18 digits where all digits can contain any
533  * value. A 32bit signed integer has a maximum of 9 digits where all digits
534  * can contain any value. 32bit integers are strongly discouraged as the possibility for
535  * conflict is much higher and would limit the total number of possible rows
536  * considerably (max 99,000 rows per day, but conflicts will be highly probable
537  * over a couple of thousand rows per day).
538  *
539  * The current formats based on length are listed below:
540  * - 17-18+ digits : YYYJJJSSSSS{6-7R}
541  * - 16 digits : YYJJJSSSSS{6R}
542  * - 14-15 digits : YJJJSSSSS{5-6R}
543  * - 13 digits : YJJJSSSS{6R}
544  * - 11-12 digits : YJJJSS{6-7R}
545  * - 9-10 digits : YJJJ{5-6R}
546  * - <9 : {1-8R}
547  *
548  * Where:
549  * - Y = last digit of year
550  * - YY = last 2 digits of year
551  * - YYY = last 3 digits of year
552  * - JJJ = Julian day of year (0-365)
553  * - SSSSS = Second of day (0-86399)
554  * - SSSS = Second of day divided by 10 (0-8639) (more fine grained than alternative of HHMM 0-2359)
555  * - SS = Second of day divided by 1000 (0-86) (more fine grained than alternative of HH 0-23)
556  * - {\#R},{\#-\#R} = number of random digits
557  *
558  * \note A time prefix is used solely for the purpose of reducing database load
559  * by making the values as incremental as possible, while still having a
560  * random portion to avoid conflicts. These are not meant to be
561  * human-interpretable numbers, and formats may change in the future. These
562  * should essentially appear completely random to a human.
563  *
564  * \param[in] pool Initialized pool object
565  * \param[in] max_len Length of unique id to generate in digits. Valid range
566  * is 9-18. May return fewer digits only when the time
567  * prefix begins with zero.
568  * \return 64bit signed integer representation of usable unique id, or 0 on
569  * misuse.
570  */
571 M_API M_int64 M_sql_gen_timerand_id(M_sql_connpool_t *pool, size_t max_len);

618 /*! Append the SQL-server-specific row lock method to the proper point in the query
619  * to be updated by a later call within the same transaction.
620  *
621  * Row locks are intended to block conflicting select statements until the current
622  * transaction has completed. It is an optimization to assist in reducing deadlocks
623  * which force rollback and retry cycles. For some database clustering solutions,
624  * like MySQL/MariaDB with Galera, it is necessary to use to prevent lost updates since
625  * updates cross-node lack the serializable isolation level guarantees.
626  *
627  * Different databases utilize different row locking methods and the methods appear
628  * at different points in the query. Due to the complexity of SQL queries, it is not
629  * viable to offer automatic rewrite ability for such queries, and instead we provide
630  * methods for simply inserting the locking statements in a DB-specific way into your
631  * query.
632  *
633  * Locking is for the duration of an SQL transaction, so row locking can only occur
634  * within a transaction, please see M_sql_trans_begin().
635  *
636  * An example query that you want to lock the rows might look like:
637  * \code SELECT * FROM "foo" WHERE "bar" = ? \endcode
638  * For a row lock for Microsoft SQL Server, the desired query with locks would look like:
639  * \code SELECT * FROM "foo" WITH (ROWLOCK, XLOCK, HOLDLOCK) WHERE "bar" = ? \endcode
640  * For the equivalent on MySQL, it would look like this:
641  * \code SELECT * FROM "foo" WHERE "bar" = ? FOR UPDATE \endcode
642  *
643  * Clearly as the above example indicates, it would be undesirable to need to rewrite
644  * the query manually by detecting the database in use, using helpers makes this
645  * easier so you do not need to have SQL-server-specific logic in your own code. Converting
646  * that query above using the helpers could be done as the below:
647  * \code{.c}
648  * M_sql_stmt_t *stmt = M_sql_stmt_create();
649  * M_buf_t *query = M_buf_create();
650  * M_sql_error_t err;
651  *
652  * M_buf_add_str(query, "SELECT * FROM \"foo\"");
653  * M_sql_query_append_updlock(pool, query, M_SQL_QUERY_UPDLOCK_TABLE);
654  * M_buf_add_str(query, " WHERE \"bar\" = ?");
655  * M_sql_stmt_bind_int32(stmt, 1);
656  * M_sql_query_append_updlock(pool, query, M_SQL_QUERY_UPDLOCK_QUERYEND);
657  * M_sql_stmt_prepare_buf(stmt, query);
658  * err = M_sql_stmt_execute(pool, stmt);
659  * //...
660  * M_sql_stmt_destroy(stmt);
661  * \endcode
662  *
663  * \note At least one #M_SQL_QUERY_UPDLOCK_TABLE must be appended per query. They will
664  * be appended immediately after each table reference (SELECT FROM ... table, or
665  * JOIN table). For the same query, at the end of the query, #M_SQL_QUERY_UPDLOCK_QUERYEND
666  * must be appended.
667  *
668  * \warning Not all databases support row-lock hints and instead rely on consistency
669  * guarantees by the underlying database for the isolation method in use. If you
670  * need these guarantees, please ensure you are using the #M_SQL_ISOLATION_SERIALIZABLE
671  * isolation method as well.
672  *
673  * \param[in] pool Initialized #M_sql_connpool_t object
674  * \param[in,out] query A pointer to an already populated M_buf_t with a partial (or complete
675  * for #M_SQL_QUERY_UPDLOCK_QUERYEND) request.
676  * \param[in] type Type of sql-specific lock to append to the query.
677  */
678 M_API void M_sql_query_append_updlock(M_sql_connpool_t *pool, M_buf_t *query, M_sql_query_updlock_type_t type);

688 /*! Perform a bitwise operation on the database using an SQL-server-specific format.
689  *
690  * A classic bitwise operation checking to see if a bit is set may look like the below:
691  * \code (exp1 & exp2) != 0 \endcode
692  *
693  * Or a bitwise operation, setting a bit or set of bits may look like:
694  * \code exp1 = exp1 | exp2 \endcode
695  *
696  * Some database servers take the expressions listed above exactly, however, others
697  * may require functions like BITAND() and BITOR() to accomplish the same thing.
698  *
699  * Taking an example of selecting all rows where "bar" has bit 4 (0x8) set:
700  * \code SELECT * FROM "foo" WHERE ("bar" & 8) <> 0 \endcode
701  * Might look like this:
702  * \code{.c}
703  * M_buf_t *buf = M_buf_create();
704  * M_sql_stmt_t *stmt = M_sql_stmt_create();
705  * M_sql_error_t err;
706  *
707  * M_buf_add_str(buf, "SELECT * FROM \"foo\" WHERE (");
708  * M_sql_query_append_bitop(pool, buf, M_SQL_BITOP_AND, "\"bar\"", "?");
709  * M_sql_stmt_bind_int32(stmt, 8);
710  * M_buf_add_str(buf, ") <> 0");
711  * M_sql_stmt_prepare_buf(stmt, buf);
712  * err = M_sql_stmt_execute(pool, stmt);
713  * //...
714  * M_sql_stmt_destroy(stmt);
715  * \endcode
716  *
717  * Of course, more complex things are possible as well, such as unsetting bits and
718  * setting others in a single request by embedding operations within eachother. Take
719  * the below example that keeps bits 2 (0x2) and 3 (0x4) while clearing the rest and
720  * also sets bit 4 (0x8):
721  * \code UPDATE "foo" SET "bar" = ( "bar" & 6 ) | 8; \endcode
722  * Might look like this:
723  * \code{.c}
724  * M_buf_t *buf = M_buf_create();
725  * M_sql_stmt_t *stmt = M_sql_stmt_create();
726  * M_sql_error_t err;
727  *
728  * M_buf_add_str(buf, "UPDATE \"foo\" SET \"bar\" = ");
729  *
730  * // Do inner-first ( "bar" & 6 )
731  * M_buf_t *inner = M_buf_create();
732  * M_buf_add_str(inner, "( ");
733  * M_sql_query_append_bitop(pool, inner, M_SQL_BITOP_AND, "\"bar\"", "?");
734  * M_sql_stmt_bind_int32(stmt, 6);
735  * M_buf_add_str(inner, " )");
736  *
737  * // Do outer, embedding inner
738  * M_sql_query_append_bitop(pool, buf, M_SQL_BITOP_OR, M_buf_peek(inner), "?");
739  * M_sql_stmt_bind_int32(stmt, 8);
740  * M_buf_cancel(inner); // We peeked, throw it away
741  *
742  * M_sql_stmt_prepare_buf(stmt, buf);
743  * err = M_sql_stmt_execute(pool, stmt);
744  * //...
745  * M_sql_stmt_destroy(stmt);
746  * \endcode
747  *
748  * \warning Most databases do not allow bitwise operations to be used for 'truth' values
749  * (e.g as a boolean). Instead, an integrator should compare the result to 0
750  * to turn it into a boolean operation if needed.
751  *
752  * \param[in] pool Initialized #M_sql_connpool_t object
753  * \param[in,out] query A pointer to an already populated M_buf_t with a partial request.
754  * \param[in] op Bitwise operation to perform.
755  * \param[in] exp1 Left-hand side of SQL expression.
756  * \param[in] exp2 Right-hande size of SQL expression.
757  * \return M_TRUE on success, M_FALSE on misuse
758  */
759 M_API M_bool M_sql_query_append_bitop(M_sql_connpool_t *pool, M_buf_t *query, M_sql_query_bitop_t op, const char *exp1, const char *exp2);
