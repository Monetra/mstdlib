38 /*! \addtogroup m_json JSON
39  * \ingroup m_formats
40  *
41  * Mostly EMCA-404 compliant JSON manipulation.
42  *
43  * Not supported:
44  * - Unicode escape conversion.
45  *
46  * Additional Features:
47  * - Comments (C/C++)
48  *
49  * Also supports most of Stefan GÃ¶ssner's JSONPath for searching.
50  * Not support are features considered redundant or potential
51  * security risks (script expressions).
52  *
53  * Example:
54  *
55  * \code{.c}
56  * M_json_node_t *j;
57  * M_json_node_t **n;
58  * size_t num_matches;
59  * size_t i;
60  * const char *s = "{ \"a\" :\n[1, \"abc\",2 ]\n}";
61  * char *out;
62  *
63  * j = M_json_read(s, M_str_len(s), M_JSON_READER_NONE, NULL, NULL, NULL, NULL);
64  * if (j == NULL) {
65  * M_printf("Could not parse json\n");
66  * return M_FALSE;
67  * }
68  *
69  * M_json_object_insert_string(j, "b", "string");
70  *
71  * n = M_json_jsonpath(j, "$.a[1::3]", &num_matches);
72  * for (i=0; i<num_matches; i++) {
73  * if (M_json_node_type(n[i]) == M_JSON_TYPE_STRING) {
74  * M_printf("%s\n", M_json_get_string(n);
75  * }
76  * }
77  * M_free(n);
78  *
79  * out = M_json_write(j, M_JSON_WRITER_PRETTYPRINT_SPACE, NULL);
80  * M_printf(out=\n%s\n", out);
81  * M_free(out);
82  *
83  * M_json_node_destroy(j);
84  * \endcode
85  *
86  * @{
87  */

95 /*! Types of JSON nodes. */
96 typedef enum {
97  M_JSON_TYPE_UNKNOWN = 0, /*!< An invalid node type. */
98  M_JSON_TYPE_OBJECT, /*!< Object (hashtable). */
99  M_JSON_TYPE_ARRAY, /*!< Array (list). */
100  M_JSON_TYPE_STRING, /*!< String. */
101  M_JSON_TYPE_INTEGER, /*!< Number. */
102  M_JSON_TYPE_DECIMAL, /*!< Floating point number. */
103  M_JSON_TYPE_BOOL, /*!< Boolean. */
104  M_JSON_TYPE_NULL /*!< JSON null type. */
105 } M_json_type_t;
106 
107 
108 /*! Flags to control the behavior of the JSON reader. */
109 typedef enum {
110  M_JSON_READER_NONE = 0, /*!< Normal operation. Treat decimal truncation as error and
111  ignore comments. */
112  M_JSON_READER_ALLOW_DECIMAL_TRUNCATION = 1 << 0, /*!< Allow decimal truncation. A decimal read and truncated will
113  not be treated as an error. */
114  M_JSON_READER_DISALLOW_COMMENTS = 1 << 1, /*!< Treat comments as an error. */
115  M_JSON_READER_OBJECT_UNIQUE_KEYS = 1 << 2 /*!< Return a parse error when an object has repeating keys. By
116  default the later key in the object will be the one used and
117  earlier keys ignored. This requires all keys in the object to
118  be unique. */
119 } M_json_reader_flags_t;
120 
121 
122 /*! Flags to control the behavior of the JSON writer. */
123 typedef enum {
124  M_JSON_WRITER_NONE = 0, /*!< No indent. All data on a single line. */
125  M_JSON_WRITER_PRETTYPRINT_SPACE = 1 << 0, /*!< 2 space indent. */
126  M_JSON_WRITER_PRETTYPRINT_TAB = 1 << 1, /*!< Tab indent. */
127  M_JSON_WRITER_PRETTYPRINT_WINLINEEND = 1 << 2 /*!< Windows line ending "\r\n" instead of Unix line ending "\n".
128  Requires space or tab pretty printing. */
129 } M_json_writer_flags_t;

136  M_JSON_ERROR_MISUSE, /*!< API missuse */

139  M_JSON_ERROR_MISSING_COMMENT_CLOSE, /*!< close comment not found */
140  M_JSON_ERROR_UNEXPECTED_COMMENT_START, /*!< unexpected / */
141  M_JSON_ERROR_INVALID_PAIR_START, /*!< expected string as first half of pair */
142  M_JSON_ERROR_DUPLICATE_KEY, /*!< duplicate key */
143  M_JSON_ERROR_MISSING_PAIR_SEPARATOR, /*!< expected ':' separator in pair */
144  M_JSON_ERROR_OBJECT_UNEXPECTED_CHAR, /*!< unexpected character in object */
145  M_JSON_ERROR_EXPECTED_VALUE, /*!< expected value after ',' */
146  M_JSON_ERROR_UNCLOSED_OBJECT, /*!< expected '}' to close object */
147  M_JSON_ERROR_ARRAY_UNEXPECTED_CHAR, /*!< unexpected character in array */
148  M_JSON_ERROR_UNCLOSED_ARRAY, /*!< expected ']' to close array */
149  M_JSON_ERROR_UNEXPECTED_NEWLINE, /*!< unexpected newline */
150  M_JSON_ERROR_UNEXPECTED_CONTROL_CHAR, /*!< unexpected control character */
151  M_JSON_ERROR_INVALID_UNICODE_ESACPE, /*!< invalid unicode escape */
152  M_JSON_ERROR_UNEXPECTED_ESCAPE, /*!< unexpected escape */
153  M_JSON_ERROR_UNCLOSED_STRING, /*!< unclosed string */
154  M_JSON_ERROR_INVALID_BOOL, /*!< invalid bool value */
155  M_JSON_ERROR_INVALID_NULL, /*!< invalid null value */
156  M_JSON_ERROR_INVALID_NUMBER, /*!< invalid number value */
157  M_JSON_ERROR_UNEXPECTED_TERMINATION, /*!< unexpected termination of string data. \0 in data. */
158  M_JSON_ERROR_INVALID_IDENTIFIER, /*!< invalid identifier */
159  M_JSON_ERROR_UNEXPECTED_END /*!< unexpected end of data */
160 } M_json_error_t;

262 /*! Using JSONPath expressions, scan for matches.
263  *
264  * Note: that full JSONPath support does not yet exist.
265  *
266  * Search expressions must start with $. They can use . to refer to the first element
267  * or .. to search for the first matching element.
268  *
269  * Supports:
270  * - Patterns containing ".", "*", "..".
271  * - Array offsets using [*]/[]/[,]/[start:end:step].
272  * - Positive offsets [0], [0,2].
273  * - Negative offsets [-1] (last item). [-2] (second to last item).
274  * - Positive and negative steps. [0:4:2]. [4:0:-1].
275  * - When counting up start is inclusive and end is exclusive. [0:3] is equivalent to [0,1,2].
276  * - When counting down start is exclusive and end is inclusive. [3:0:-1] is equivalent to [2,1,0].
277  *
278  * Does not Support:
279  * - Braket notation ['x'].
280  * - Filter/script expressions. [?(exp)]/[(exp)].
281  *
282  * \param[in] node The node.
283  * \param[in] search search expression
284  * \param[out] num_matches Number of matches found
285  *
286  * \return array of M_json_node_t pointers on success (must free array, but not internal pointers), NULL on failure
287  *
288  * \see M_free
289  */
290 M_API M_json_node_t **M_json_jsonpath(const M_json_node_t *node, const char *search, size_t *num_matches) M_MALLOC;
