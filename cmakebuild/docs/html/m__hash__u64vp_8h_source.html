36 /*! \addtogroup m_hash_u64vp Hashtable - uint64/Void Pointer

39  * Hashtable, meant for storing uint64 keys and void pointer values.

60 /*! Flags for controlling the behavior of the hash */
61 typedef enum {
62  M_HASH_U64VP_NONE = 0, /*!< Case sensitive single value (new values replace). */
63  M_HASH_U64VP_KEYS_ORDERED = 1 << 0, /*!< Keys should be ordered. Default is insertion order unless the
64  sorted option is specified. */
65  M_HASH_U64VP_KEYS_SORTASC = 1 << 1, /*!< When the keys are ordered sort them using the key_equality function. */
66  M_HASH_U64VP_KEYS_SORTDESC = 1 << 2, /*!< When the keys are ordered sort them using the key_equality function. */
67  M_HASH_U64VP_MULTI_VALUE = 1 << 3, /*!< Allow keys to contain multiple values.
68  Sorted in insertion order another sorting is specified. */
69  M_HASH_U64VP_MULTI_GETLAST = 1 << 4 /*!< When using get and get_direct function get the last value from the list
70  when allowing multiple values. The default is to get the first value. */
71 } M_hash_u64vp_flags_t;

76 /*! Create a new hashtable.
77  *
78  * The hashtable will pre-allocate an array of buckets based on the rounded up size specified. Any hash collisions
79  * will result in those collisions being chained together via a linked list. The hashtable will auto-expand by a
80  * power of 2 when the fill percentage specified is reached. All key entries are compared in a case-insensitive
81  * fashion, and are duplicated internally. Values are duplicated. Case is preserved for both keys and values.
82  *
83  * \param[in] size Size of the hash table. If not specified as a power of 2, will
84  * be rounded up to the nearest power of 2.
85  * \param[in] fillpct The maximum fill percentage before the hash table is expanded. If
86  * 0 is specified, the hashtable will never expand, otherwise the
87  * value must be between 1 and 99 (recommended: 75).
88  * \param[in] flags M_hash_strvp_flags_t flags for modifying behavior.
89  * \param[in] destroy_func The function to be called to destroy value when the hashtable
90  * itself is destroyed. Can be NULL.
91  *
92  * \return Allocated hashtable.
93  *
94  * \see M_hash_u64vp_destroy
95  */
96 M_API M_hash_u64vp_t *M_hash_u64vp_create(size_t size, M_uint8 fillpct, M_uint32 flags, void (*destroy_func)(void *)) M_MALLOC_ALIASED;

285 /*! Merge two hashtables together.
286  *
287  * The second (src) hashtable will be destroyed automatically upon completion of this function. Any key/value
288  * pointers for the hashtable will be directly copied over to the destination hashtable, they will not be
289  * duplicated. Any keys which exist in 'dest' that also exist in 'src' will be overwritten by the 'src' value.
290  *
291  * \param[in,out] dest Pointer by reference to the hashtable receiving the key/value pairs.
292  * if dest is NULL, the src address will simply be copied to dest.
293  * \param[in,out] src Pointer to the hashtable giving up its key/value pairs.
294  */
295 M_API void M_hash_u64vp_merge(M_hash_u64vp_t **dest, M_hash_u64vp_t *src) M_FREE(2);
