36 /*! \addtogroup m_thread_common Common Threading Functions
37  * \ingroup m_thread
38  *
39  * Thread handling.
40  *
41  * System specific threading model as well as a cooperative threading model
42  * is available. Cooperative should only be used on system that do not natively
43  * support threads. Such as some embedded systems.
44  *
45  * By default threads are created in a detached state. M_thread_attr_t must be
46  * used in order to have a thread created in a joinable state.
47  *
48  * Example:
49  *
50  * \code{.c}
51  * static M_uint32 count = 0;
52  *
53  * static void td(void)
54  * {
55  * M_printf("Thread finished\n");
56  * }
57  *
58  * static void *runner(void *arg)
59  * {
60  * M_thread_mutex_t *m = arg;
61  *
62  * M_thread_mutex_lock(m);
63  * count++;
64  * M_thread_mutex_unlock(m);
65  * }
66  *
67  * int main(int argc, char **argv)
68  * {
69  * M_threadid_t t1;
70  * M_threadid_t t2;
71  * M_thread_attr_t *tattr;
72  * M_thread_mutex_t *m
73  *
74  * M_thread_destructor_insert(td);
75  *
76  * tattr = M_thread_attr_create();
77  * M_thread_attr_set_create_joinable(tattr, M_TRUE);
78  *
79  * m = M_thread_mutex_create(M_THREAD_MUTEXATTR_NONE);
80  *
81  * t1 = M_thread_create(tattr, runner, m);
82  * t2 = M_thread_create(tattr, runner, m);
83  *
84  * M_thread_attr_destroy(tattr);
85  * M_thread_join(t1, NULL);
86  * M_thread_join(t2, NULL);
87  *
88  * M_thread_mutex_destroy(m);
89  *
90  * M_printf("count='%u'\n", count);
91  *
92  * return 0;
93  * }
94  * \endcode
95  *
96  */
97

142 /*! Adds a function to be called each time a thread finishes.
143  *

663 /*! Lock a spinlock.
664  *
665  * A spinlock is similar in usage to a mutex, but should NOT be used in place of a mutex.
666  * When in doubt, use a mutex instead, a spinlock is almost always the wrong thing to use.
667  * Spinlocks can be used protect areas of memory that are very unlikely to have high
668  * contention and should only be held for very short durations, or when the act of
669  * initializing a mutex might itself cause a race condition (such as during an initialization
670  * procedure as mutexes do not support static initializers).
671  *
672  * When lock contention occurs on a spinlock, it will spin, consuming CPU, waiting
673  * for the lock to be released. Spinlocks are purely implemented in userland
674  * using atomics. The implementation uses 'tickets' to try to guarantee lock order
675  * in a first-come first-served manner, and has rudimentary backoff logic to attempt
676  * to reduce resource consumption during periods of high lock contention.
677  *
678  * A spinlock variable must have been initialized using M_THREAD_SPINLOCK_STATIC_INITIALIZER
679  * and passed by reference into this function. There is no initialization or
680  * destruction function.
681  *
682  * \param[in] spinlock Spinlock initialized via M_THREAD_SPINLOCK_STATIC_INITIALIZER
683  * and passed by reference.
684  */
685 M_API void M_thread_spinlock_lock(M_thread_spinlock_t *spinlock);
686 
687 
688 /*! Unlock a spinlock
689  *
690  * See M_thread_spinlock_lock() for more information.
691  *
692  * \param[in] spinlock Spinlock initialized via M_THREAD_SPINLOCK_STATIC_INITIALIZER
693  * and passed by reference.
694  */
695 M_API void M_thread_spinlock_unlock(M_thread_spinlock_t *spinlock);

719 /*! Ensure an initialization routine is performed only once, even if called from
720  * multiple threads simultaneously.
721  *
722  * Performing initialization in a multi-threaded program can cause race conditions.
723  *
724  * Take this code example:
725  * static int initialized = 0;
726  * if (!initialized) {
727  * init_routine();
728  * initialized = 1;
729  * }
730  *
731  * If two threads where to enter this simultaneously, before init_routine() was
732  * complete, they would call it twice. The above code example can be replaced
733  * with:
734  *
735  * static M_thread_once_t initialized = M_THREAD_ONCE_STATIC_INITIALIZER;
736  * M_thread_once(&initialized, init_routine);
737  *
738  * \param[in] once_control Once control variable passed by reference, and first set
739  * to M_THREAD_ONCE_STATIC_INITIALIZER;
740  * \param[in] init_routine Initialization routine to be called if it has not yet
741  * been called.
742  * \param[in] init_flags Flags to be passed onto the initialization routine.
743  * \return M_TRUE if init routine was just run, M_FALSE if not run (previously run)
744  */
745 M_API M_bool M_thread_once(M_thread_once_t *once_control, void (*init_routine)(M_uint64 flags), M_uint64 init_flags);

748 /*! Reset the once_control object back to an uninitialized state. Useful to be
749  * called in a destructor so an initialization routine can be re-run.
750  *
751  * \param[in] once_control Once control variable passed by reference, and first set
752  * to M_THREAD_ONCE_STATIC_INITIALIZER;
753  * \return M_TRUE if reset, M_FALSE if not initialized.
754  */
