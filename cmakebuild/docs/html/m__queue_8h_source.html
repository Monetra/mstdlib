35 /*! \addtogroup m_queue Queue
36  * \ingroup m_datastructures
37  *
38  * Queue, meant for storing a list of pointers in either insertion order or
39  * sorted as per a user-defined callback. The usage is very similar to that of
40  * an M_llist_t, and infact an M_llist_t is used as one of the backing data types,
41  * but it also uses an M_hashtable_t in order to make lookups and removals of
42  * queue members by pointer O(1) rather than O(log n) for sorted lists
43  * or O(n) for unsorted lists.
44  *
45  * Usage Example:
46  *
47  * \code{.c}
48  * M_queue_t *queue;
49  * void *member = NULL;
50  * M_queue_foreach_t *q_foreach = NULL;
51  *
52  * queue = M_queue_create(M_sort_compar_str, M_free);
53  * M_queue_insert(queue, M_strdup("b. hello world"));
54  * M_queue_insert(queue, M_strdup("c. goodbye"));
55  * M_queue_insert(queue, M_strdup("a! -- I should end up enumerated first"));
56  * M_printf("queue members: %zu\n", M_queue_len(queue));
57  *
58  * while (M_queue_foreach(queue, &q_foreach, &member)) {
59  * M_printf("removing member: %s\n", (const char *)member);
60  * M_queue_remove(queue, member);
61  * }
62  *
63  * M_printf("queue members: %zu\n", M_queue_len(queue));
64  * M_queue_destroy(queue);
65  * \endcode
66  *
67  * @{
68  */

80 /*! Create a queue (list of objects) that stores user-provided pointers. The pointers
81  * stored may be kept in insertion order or sorted, depending on how the queue is
82  * initialized.
83  * \param sort_cb If the pointers should be stored in a sorted order, register this
84  * callback with the routine for sorting. If insertion order is
85  * desired, pass NULL.
86  * \param free_cb Upon removal of a pointer from the queue, this callback will be
87  * called. This applies to M_queue_destroy() and M_queue_remove().
88  * If no free callback desired, pass NULL.
89  * \return Allocated M_queue_t * on success that should be free'd with M_queue_destroy(),
90  * otherwise NULL.
91  */
92 M_API M_queue_t *M_queue_create(M_sort_compar_t sort_cb, void (*free_cb)(void *));

160 /*! Enumerate across all members in the queue. This function is designed to be run
161  * in a while() loop until it returns M_FALSE. The q_foreach parameter will be
162  * automatically deallocated if this returns M_FALSE, otherwise if breaking out of
163  * the loop early, M_queue_foreach_free must be called.
164  *
165  * During an enumeration, it is allowable to remove the *current* member from the
166  * queue. It is undefined behavior to remove any other member during an enumeration.
167  * Addition to the queue is also allowed during an enumeration, however it is not
168  * defined if the new value will end up in the enumerated set.
169  * \param queue Initialized queue object returned by M_queue_create()
170  * \param q_foreach M_queue_foreach_t * passed By Reference, and initialized to NULL
171  * before the first call to M_queue_foreach(). The value returned
172  * is not meant to be interpreted and must be passed back in, unmodified
173  * to the next call of M_queue_foreach(). If breaking out of the loop
174  * prior to M_queue_foreach() returning M_FALSE, this object should be
175  * free'd with M_queue_foreach_free().
176  * \param member Pointer to member to be filled in, passed By Reference. This value
177  * will be populated with the current member in the enumeration. Must
178  * not be NULL.
179  * \return M_TRUE if there are more members to enumerate, M_FALSE if no more members and no
180  * result was returned.
181  */
182 M_API M_bool M_queue_foreach(M_queue_t *queue, M_queue_foreach_t **q_foreach, void **member);
183

185 /*! Free's the M_queue_foreach_t * filled in by M_queue_foreach. This only needs to be
186  * called if the enumeration was ended early and not allowed to run to completion.
187  * NOTE: This may currently be a no-op if q_foreach just references an internal pointer.
188  * \param q_foreach M_queue_foreach_t * initialized by M_queue_foreach.
189  */
190 M_API void M_queue_foreach_free(M_queue_foreach_t *q_foreach);
