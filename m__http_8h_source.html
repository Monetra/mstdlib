190 /*! Create query string, append to given URI, return as new string.
191  *
192  * Empty values are not permitted - keys whose values are set to the empty string will be left out of
193  * the query string.
194  *
195  * Web applications use two slightly-different URL encodings for query strings, one that encodes spaces
196  * as '%20' (\link M_TEXTCODEC_PERCENT_URL M_TEXTCODEC_PERCENT_URL\endlink), and one that encodes spaces
197  * as '+' (\link M_TEXTCODEC_PERCENT_URLPLUS M_TEXTCODEC_PERCENT_URLPLUS\endlink). Web apps are about
198  * evenly split between these two options, so the caller must pick which one to use based on their own
199  * needs, by setting the \a use_plus parameter.
200  *
201  * \see M_http_add_query_string_buf()
202  *
203  * \param[in] uri uri string (e.g., /cgi-bin/payment/start, or %http://google.com/whatever)
204  * \param[in] params key-value pairs to encode in query string
205  * \param[in] use_plus if M_TRUE, sub in '+' for space character. Otherwise, use '%20'.
206  * \return new string with URI + query string, or \c NULL if there was an encoding error
207  */
208 M_API char *M_http_add_query_string(const char *uri, M_hash_dict_t *params, M_bool use_plus);
209 
210 
211 /*! Create query string, append URI + query string to buffer.
212  *
213  * Empty values are not permitted - keys whose values are set to the empty string will be left out of
214  * the query string.
215  *
216  * Web applications use two slightly-different URL encodings for query strings, one that encodes spaces
217  * as '%20' (\link M_TEXTCODEC_PERCENT_URL M_TEXTCODEC_PERCENT_URL\endlink), and one that encodes spaces
218  * as '+' (\link M_TEXTCODEC_PERCENT_URLPLUS M_TEXTCODEC_PERCENT_URLPLUS\endlink). Web apps are about
219  * evenly split between these two options, so the caller must pick which one to use based on their own
220  * needs, by setting the \a use_plus parameter.
221  *
222  * \see M_http_add_query_string()
223  *
224  * \param[out] buf buffer to add URI + query string to, contents remain unchanged if there was an error
225  * \param[in] uri uri string (e.g., /cgi-bin/payment/start, or %http://google.com/whatever)
226  * \param[in] params key-value pairs to encode in query string
227  * \param[in] use_plus if M_TRUE sub in '+' for space character, otherwise will use '%20'
228  * \return M_TRUE if successful, or \c M_FALSE if there was an encoding error
229  */
230 M_API M_bool M_http_add_query_string_buf(M_buf_t *buf, const char *uri, M_hash_dict_t *params, M_bool use_plus);
