79  * M_hash_dict_enumerate(headers, &he);

94  * M_event_done(el);
95  * }

104  * io_type = "READ";
105  * break;
106  * case M_IO_TRACE_TYPE_WRITE:
107  * io_type = "WRITE";

109  * case M_IO_TRACE_TYPE_EVENT:
