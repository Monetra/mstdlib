47 /*! Error codes. */
48 typedef enum {
49  M_UTF8_ERROR_SUCCESS, /*!< Success. */
50  M_UTF8_ERROR_BAD_START, /*!< Start of byte sequence is invalid. */
51  M_UTF8_ERROR_TRUNCATED, /*!< The utf-8 character length exceeds the data length. */

80 /*! Ge the number of utf-8 characters in a string.
81  *
82  * This is the number of characters not the number of bytes in the string.
83  * M_str_len will only return the same value if the string is only ascii.
84  *
85  * \param[in] str utf-8 string.
86  *
87  * \return Number of characters on success. On failure will return 0. Use
88  * M_str_isempty to determine if 0 is a failure or empty string.
89  */
90 M_API size_t M_utf8_cnt(const char *str);
